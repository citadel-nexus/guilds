Upgrade the Citadel Nexus email system (`/admin/operations` → Email tab) with:
1. **AI-generated email responses** (OpenAI / Bedrock)
1. **Dropdown template selector** with live preview
1. **Client data rehydration** from Supabase (auto-fill recipient info)
1. **Header/footer injection** with sender identity (email, socials, phone) from auto-selected template
1. **Schedule meeting with sender** (calendar link generation)

```python
Email System Upgrade Routes — Citadel Nexus
Features: AI generation, template dropdown, client rehydration,

@router.get("/templates")
async def list_templates(category: Optional[str] = None):
    """Return all email templates, optionally filtered by category."""
    query = supabase.table("email_templates").select("*").order("name")
    if category:
        query = query.eq("category", category)
    return query.execute().data


@router.get("/templates/{slug}")
async def get_template(slug: str):
    """Fetch a single template by slug."""
    result = (
        supabase.table("email_templates")
        .select("*")
        .eq("slug", slug)
        .single()
        .execute()
    )
    if not result.data:
        raise HTTPException(404, f"Template '{slug}' not found")
    return result.data


@router.post("/templates")
async def create_template(template: EmailTemplate):
    payload = template.model_dump(exclude_none=True)
    payload.setdefault("id", str(uuid.uuid4()))
    return supabase.table("email_templates").insert(payload).execute().data


@router.put("/templates/{slug}")
async def update_template(slug: str, template: EmailTemplate):
    payload = template.model_dump(exclude_none=True)
    payload["updated_at"] = datetime.utcnow().isoformat()
    return (
        supabase.table("email_templates")
        .update(payload)
        .eq("slug", slug)
        .execute()
        .data
    )

@router.get("/rehydrate")
async def rehydrate_client(
    email: Optional[str] = None,
    name: Optional[str] = None,
    phone: Optional[str] = None,
):
    """
    Rehydrate client data from Supabase by email, name, or phone.
    Returns full profile + meeting history for template variable injection.
    """
    if not any([email, name, phone]):
        raise HTTPException(400, "Provide at least one of: email, name, phone")

    query = supabase.table("clients").select(
        "*, meetings:client_meetings(*)"
    )
    if email:
        query = query.eq("email", email)
    elif name:
        query = query.ilike("name", f"%{name}%")
    elif phone:
        query = query.eq("phone", phone)

    result = query.execute()
    if not result.data:
        return {"found": False, "client": None}

    client = result.data[0]
    return {"found": True, "client": client}


@router.get("/rehydrate/search")
async def search_clients(q: str = Query(..., min_length=2)):
    """Fuzzy search for the To: autocomplete dropdown."""
    return (
        supabase.table("clients")
        .select("id, name, email, company, phone")
        .or_(f"name.ilike.%{q}%,email.ilike.%{q}%,company.ilike.%{q}%")
        .limit(10)
        .execute()
        .data
    )

@router.post("/sender-signatures")
async def upsert_sender_signature(sig: SenderSignature):
    """Create or update a sender signature; auto-generates header/footer HTML."""
    sig_dict = sig.model_dump(exclude_none=True)
    sig_dict.setdefault("id", str(uuid.uuid4()))
    sig_dict["header_html"] = _build_header(sig)
    sig_dict["footer_html"] = _build_footer(sig)

@router.post("/ai-generate")
async def ai_generate_email(req: AIGenerateRequest):
    """
    Generate email body via AI. Optionally seeds from a template for structure.
    """
    template_hint = ""
    if req.template_slug:
        tmpl = await get_template(req.template_slug)
        template_hint = (
            f"Use this template structure as a guide:\n"
            f"Subject: {tmpl['subject']}\n"
            f"Variables: {', '.join(tmpl.get('variables', []))}\n"
        )

    system_prompt = (
        "You are an email copywriter for Citadel Nexus, an AI-orchestrated "
        "development platform. "
        f"Write a {req.tone} email for the intent: {req.intent}. "
        "Keep it concise, genuine, and action-oriented. "
        "Do NOT include a subject line — only the email body in HTML. "
        "Always address the recipient by first name. "
        f"{template_hint}"
    )
    user_prompt = (
        f"Recipient: {req.recipient_name} ({req.recipient_email})\n"
        f"Topic: {req.topic}\n"
        f"Additional context: {req.context or 'None'}\n\n"
        "Write the email body now."
    )

    body = await _call_llm(system_prompt, user_prompt, req.max_tokens)
    return {
        "generated_body": body,
        "tone": req.tone,
        "intent": req.intent,
        "model_used": "openai" if OPENAI_KEY else "bedrock",
    }


async def _call_llm(system: str, user: str, max_tokens: int) -> str:
    """OpenAI (preferred) or AWS Bedrock fallback."""
    if OPENAI_KEY:
        async with httpx.AsyncClient() as client:
            resp = await client.post(
                "https://api.openai.com/v1/chat/completions",
                headers={"Authorization": f"Bearer {OPENAI_KEY}"},
                json={
                    "model": "gpt-4o-mini",
                    "messages": [
                        {"role": "system", "content": system},
                        {"role": "user", "content": user},
                    ],
                    "max_tokens": max_tokens,
                    "temperature": 0.7,
                },
                timeout=30,
            )
            resp.raise_for_status()
            return resp.json()["choices"][0]["message"]["content"]
    else:
        import boto3
        bedrock = boto3.client("bedrock-runtime", region_name=BEDROCK_REGION)
        body = json.dumps({
            "anthropic_version": "bedrock-2023-05-31",
            "messages": [{"role": "user", "content": f"{system}\n\n{user}"}],
            "max_tokens": max_tokens,
            "temperature": 0.7,
        })
        response = bedrock.invoke_model(
            modelId="anthropic.claude-3-haiku-20240307-v1:0", body=body
        )
        result = json.loads(response["body"].read())
        return result["content"][0]["text"]

         style="display:inline-block; padding:12px 32px; background:#0ff1ce;

        sig = await get_sender_signature(req.sender_signature_id)

    return {
        "meeting_id": meeting_id,
        "calendar_link": cal_link,
        "email_body": body,
        "status": "proposed",

# 6. SEND EMAIL (template + signature injection + audit log)
# ══════════════════════════════════════════════════════════

@router.post("/send")
async def send_email(req: SendEmailRequest):
    import boto3

    final_html = req.body_html
    header = ""
    footer = ""

    # Inject sender signature header/footer
    if req.sender_signature_id:
        sig = await get_sender_signature(req.sender_signature_id)
        header = sig.get("header_html", "")
        footer = sig.get("footer_html", "")
    else:
        # Use default signature
        default = (
            supabase.table("sender_signatures")
            .select("*")
            .eq("is_default", True)
            .limit(1)
            .execute()
        )
        if default.data:
            header = default.data[0].get("header_html", "")
            footer = default.data[0].get("footer_html", "")

    wrapped = f"""
    <div style="max-width:600px; margin:0 auto; font-family:sans-serif;
                background:#0a0a0a; color:#e0e0e0;">
      {header}
      <div style="padding:24px;">{final_html}</div>
      {footer}
    </div>"""

    ses = boto3.client("ses", region_name=SES_REGION)
    response = ses.send_email(
        Source=SES_FROM,
        Destination={"ToAddresses": [req.to]},
        Message={
            "Subject": {"Data": req.subject},
            "Body": {
                "Html": {"Data": wrapped},
                "Text": {"Data": _strip_html(final_html)},
            },
        },
    )

    # Audit log
    supabase.table("email_send_log").insert({
        "to_email": req.to,
        "subject": req.subject,
        "template_slug": req.template_slug,
        "signature_id": req.sender_signature_id,
        "ai_generated": False,
        "status": "sent",
        "ses_message_id": response.get("MessageId", ""),
    }).execute()

    return {"status": "sent", "message_id": response.get("MessageId")}


def _strip_html(html: str) -> str:
    """Naive HTML stripper for plain-text fallback."""
    import re
    return re.sub(r"<[^>]+>", "", html).strip()
```

## Frontend — React Component

File: `src/components/EmailCompose.jsx`
```javascript
/**
 * EmailCompose.jsx — Citadel Nexus Email System
 * Features: template dropdown, AI generate, client rehydration,
 *           sender signature, meeting scheduler, live preview
 */
import React, { useState, useEffect, useCallback } from "react";
import debounce from "lodash.debounce";

const API = "/api/email";

export default function EmailCompose() {
  // ── State ─────────────────────────────────────────
  const [templates, setTemplates] = useState([]);
  const [selectedTemplate, setSelectedTemplate] = useState(null);
  const [signatures, setSignatures] = useState([]);
  const [selectedSignature, setSelectedSignature] = useState(null);

  const [to, setTo] = useState("");
  const [toSuggestions, setToSuggestions] = useState([]);
  const [clientData, setClientData] = useState(null);

  const [subject, setSubject] = useState("");
  const [bodyHtml, setBodyHtml] = useState("");
  const [topic, setTopic] = useState("");
  const [nextSteps, setNextSteps] = useState("");

  const [aiTone, setAiTone] = useState("professional");
  const [aiIntent, setAiIntent] = useState("follow_up");
  const [aiLoading, setAiLoading] = useState(false);

  const [meetingTopic, setMeetingTopic] = useState("");
  const [meetingDuration, setMeetingDuration] = useState(30);
  const [proposedTimes, setProposedTimes] = useState([""]);
  const [showMeetingPanel, setShowMeetingPanel] = useState(false);

  const [sending, setSending] = useState(false);
  const [sendResult, setSendResult] = useState(null);

  // ── Load templates + signatures on mount ──────────
  useEffect(() => {
    fetch(`${API}/templates`)
      .then((r) => r.json())
      .then(setTemplates);
    fetch(`${API}/sender-signatures`)
      .then((r) => r.json())
      .then((sigs) => {
        setSignatures(sigs);
        const def = sigs.find((s) => s.is_default);
        if (def) setSelectedSignature(def);
      });
  }, []);

  // ── Template selection ────────────────────────────
  const onTemplateSelect = async (slug) => {
    if (!slug) {
      setSelectedTemplate(null);
      setSubject("");
      setBodyHtml("");
      return;
    }
    const res = await fetch(`${API}/templates/${slug}`);
    const tmpl = await res.json();
    setSelectedTemplate(tmpl);
    setSubject(tmpl.subject);
    setBodyHtml(tmpl.body_html);
  };

  // ── Client rehydration (To: field autocomplete) ───
  const searchClients = useCallback(
    debounce(async (q) => {
      if (q.length < 2) return setToSuggestions([]);
      const res = await fetch(
        `${API}/rehydrate/search?q=${encodeURIComponent(q)}`
      );
      setToSuggestions(await res.json());
    }, 300),
    []
  );

  const onSelectClient = async (client) => {
    setTo(client.email);
    setToSuggestions([]);
    // Full rehydration
    const res = await fetch(
      `${API}/rehydrate?email=${encodeURIComponent(client.email)}`
    );
    const data = await res.json();
    if (data.found) {
      setClientData(data.client);
      // Inject client name into template variables
      if (bodyHtml.includes("recipient_name")) {
        setBodyHtml(
          bodyHtml.replace(/\{\{recipient_name\}\}/g, data.client.name)
        );
      }
      if (subject.includes("recipient_name")) {
        setSubject(
          subject.replace(/\{\{recipient_name\}\}/g, data.client.name)
        );
      }
    }
  };

  // ── AI Generation ─────────────────────────────────
  const handleAIGenerate = async () => {
    setAiLoading(true);
    try {
      const res = await fetch(`${API}/ai-generate`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          template_slug: selectedTemplate?.slug || null,
          recipient_name: clientData?.name || to.split("@")[0],
          recipient_email: to,
          topic: topic || subject,
          context: nextSteps || null,
          tone: aiTone,
          intent: aiIntent,
        }),
      });
      const data = await res.json();
      setBodyHtml(data.generated_body);
    } catch (err) {
      console.error("AI generation failed:", err);
    } finally {
      setAiLoading(false);
    }
  };

  // ── Schedule Meeting ──────────────────────────────
  const handleScheduleMeeting = async () => {
    const res = await fetch(`${API}/schedule-meeting`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        recipient_name: clientData?.name || to.split("@")[0],
        recipient_email: to,
        proposed_times: proposedTimes.filter(Boolean),
        duration_minutes: meetingDuration,
        topic: meetingTopic || topic || "Follow-up call",
        sender_signature_id: selectedSignature?.id || null,
      }),
    });
    const data = await res.json();
    setBodyHtml(data.email_body);
    setSubject(`Meeting: ${meetingTopic || topic || "Follow-up"}`);
    setShowMeetingPanel(false);
  };

  // ── Send ───────────────────────────────────────────
  const handleSend = async () => {
    setSending(true);
    setSendResult(null);
    try {
      const res = await fetch(`${API}/send`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          to,
          subject,
          body_html: bodyHtml,
          sender_signature_id: selectedSignature?.id || null,
          template_slug: selectedTemplate?.slug || null,
        }),
      });
      const data = await res.json();
      setSendResult(data);
    } catch (err) {
      setSendResult({ status: "error", message: err.message });
    } finally {
      setSending(false);
    }
  };

  // ── Variable injection helper ─────────────────────
  const injectVariables = (html) => {
    if (!html) return html;
    let result = html;
    if (clientData) {
      result = result
        .replace(/\{\{recipient_name\}\}/g, clientData.name || "")
        .replace(/\{\{company\}\}/g, clientData.company || "")
        .replace(/\{\{phone\}\}/g, clientData.phone || "");
    }
    result = result
      .replace(/\{\{topic\}\}/g, topic || "")
      .replace(/\{\{next_steps\}\}/g, nextSteps || "")
      .replace(
        /\{\{sender_name\}\}/g,
        selectedSignature?.sender_name || "Citadel Nexus"
      );
    return result;
  };

  // ── Render ─────────────────────────────────────────
  return (
    <div className="email-compose" style={styles.container}>
      {/* ── Header row: Template + Sender dropdown ── */}
      <div style={styles.row}>
        <label style={styles.label}>Template</label>
        <select
          style={styles.select}
          onChange={(e) => onTemplateSelect(e.target.value)}
          value={selectedTemplate?.slug || ""}
        >
          <option value="">— Select Template —</option>
          {templates.map((t) => (
            <option key={t.slug} value={t.slug}>
              {t.name} ({t.category})
            </option>
          ))}
        </select>

        <label style= ...styles.label, marginLeft: 16 >From</label>
        <select
          style={styles.select}
          onChange={(e) => {
            const sig = signatures.find((s) => s.id === e.target.value);
            setSelectedSignature(sig || null);
          }}
          value={selectedSignature?.id || ""}
        >
          {signatures.map((s) => (
            <option key={s.id} value={s.id}>
              {s.sender_name} ({s.sender_email})
            </option>
          ))}
        </select>
      </div>

      {/* ── To: with autocomplete + rehydration ── */}
      <div style={styles.row}>
        <label style={styles.label}>To</label>
        <div style= position: "relative", flex: 1 >
          <input
            style={styles.input}
            placeholder="Name or email@example.com"
            value={to}
            onChange={(e) => {
              setTo(e.target.value);
              searchClients(e.target.value);
            }}
          />
          {toSuggestions.length > 0 && (
            <div style={styles.dropdown}>
              {toSuggestions.map((c) => (
                <div
                  key={c.id}
                  style={styles.dropdownItem}
                  onClick={() => onSelectClient(c)}
                >
                  <strong>{c.name}</strong> — {c.email}
                  {c.company && ` (${c.company})`}
                </div>
              ))}
            </div>
          )}
        </div>
        {clientData && (
          <span style={styles.rehydratedBadge}>
            ✅ {clientData.name}
            {clientData.company && ` · ${clientData.company}`}
          </span>
        )}
      </div>

      {/* ── Subject ── */}
      <div style={styles.row}>
        <label style={styles.label}>Subject</label>
        <input
          style={styles.input}
          value={subject}
          onChange={(e) => setSubject(e.target.value)}
        />
      </div>

      {/* ── Topic + Next Steps (template variables) ── */}
      <div style={styles.row}>
        <label style={styles.label}>Topic</label>
        <input
          style={styles.input}
          placeholder="AI workflow demo"
          value={topic}
          onChange={(e) => setTopic(e.target.value)}
        />
      </div>
      <div style={styles.row}>
        <label style={styles.label}>Next Steps</label>
        <input
          style={styles.input}
          placeholder="We'll schedule a kickoff call..."
          value={nextSteps}
          onChange={(e) => setNextSteps(e.target.value)}
        />
      </div>

      {/* ── AI Generation controls ── */}
      <div style={styles.row}>
        <select
          style={styles.selectSmall}
          value={aiTone}
          onChange={(e) => setAiTone(e.target.value)}
        >
          <option value="professional">Professional</option>
          <option value="friendly">Friendly</option>
          <option value="formal">Formal</option>
          <option value="casual">Casual</option>
        </select>
        <select
          style={styles.selectSmall}
          value={aiIntent}
          onChange={(e) => setAiIntent(e.target.value)}
        >
          <option value="follow_up">Follow Up</option>
          <option value="intro">Introduction</option>
          <option value="proposal">Proposal</option>
          <option value="thank_you">Thank You</option>
          <option value="schedule">Schedule</option>
        </select>
        <button
          style={styles.aiButton}
          onClick={handleAIGenerate}
          disabled={aiLoading || !to}
        >
          {aiLoading ? "⏳ Generating..." : "✨ AI Generate"}
        </button>
        <button
          style={styles.meetingButton}
          onClick={() => setShowMeetingPanel(!showMeetingPanel)}
        >
          📅 Schedule Meeting
        </button>
      </div>

      {/* ── Meeting scheduler panel ── */}
      {showMeetingPanel && (
        <div style={styles.meetingPanel}>
          <h4 style= margin: "0 0 8px", color: "#0ff1ce" >
            📅 Schedule Meeting
          </h4>
          <input
            style={styles.input}
            placeholder="Meeting topic"
            value={meetingTopic}
            onChange={(e) => setMeetingTopic(e.target.value)}
          />
          <div style= display: "flex", gap: 8, marginTop: 8 >
            <select
              style={styles.selectSmall}
              value={meetingDuration}
              onChange={(e) => setMeetingDuration(+e.target.value)}
            >
              <option value={15}>15 min</option>
              <option value={30}>30 min</option>
              <option value={45}>45 min</option>
              <option value={60}>60 min</option>
            </select>
            {proposedTimes.map((t, i) => (
              <input
                key={i}
                type="datetime-local"
                style={styles.input}
                value={t}
                onChange={(e) => {
                  const copy = [...proposedTimes];
                  copy[i] = e.target.value;
                  setProposedTimes(copy);
                }}
              />
            ))}
            <button
              style={styles.smallBtn}
              onClick={() => setProposedTimes([...proposedTimes, ""])}
            >
              + Time
            </button>
          </div>
          <button
            style= ...styles.aiButton, marginTop: 8 
            onClick={handleScheduleMeeting}
          >
            Generate Meeting Email
          </button>
        </div>
      )}

      {/* ── Editor ── */}
      <div style={styles.row}>
        <label style={styles.label}>Body</label>
        <textarea
          style={styles.textarea}
          rows={12}
          value={bodyHtml}
          onChange={(e) => setBodyHtml(e.target.value)}
        />
      </div>

      {/* ── Action buttons ── */}
      <div style={styles.row}>
        <button
          style={styles.sendButton}
          onClick={handleSend}
          disabled={sending || !to || !subject}
        >
          {sending ? "Sending..." : "📤 Send"}
        </button>
        <button
          style={styles.previewButton}
          onClick={() =>
            window.open(
              "about:blank"
            )
          }
        >
          👁 Preview
        </button>
      </div>

      {sendResult && (
        <div
          style=
            ...styles.result,
            color: sendResult.status === "sent" ? "#0ff1ce" : "#ff4444",
          
        >
          {sendResult.status === "sent"
            ? `✅ Sent! Message ID: ${sendResult.message_id}`
            : `❌ Error: ${sendResult.message}`}
        </div>
      )}

      {/* ── Live Preview Pane ── */}
      <div style={styles.previewPane}>
        <h4 style= color: "#888", margin: "0 0 8px" >Preview</h4>
        <div
          style={styles.previewFrame}
          dangerouslySetInnerHTML={{
            __html: `
              <div style="max-width:600px; margin:0 auto; background:#0a0a0a;
                          color:#e0e0e0; font-family:sans-serif;">
                ${selectedSignature?.header_html || ""}
                <div style="padding:24px;">
                  ${injectVariables(bodyHtml)}
                </div>
                ${selectedSignature?.footer_html || ""}
              </div>
            `,
          }}
        />
      </div>
    </div>
  );
}
```

## Feature Summary

| **Feature** | **Backend Endpoint** | **Frontend Hook** | **Data Source** |
| Template Dropdown | `GET /api/email/templates` | `onTemplateSelect()` | Supabase `email_templates` |
| AI Generation | `POST /api/email/ai-generate` | `handleAIGenerate()` | OpenAI / Bedrock |
| Client Rehydration | `GET /api/email/rehydrate` | `onSelectClient()` | Supabase `clients` |
| Sender Signature | `GET /api/email/sender-signatures` | From dropdown | Supabase `sender_signatures` |
| Header/Footer Inject | Auto on `POST /send` | Live preview pane | Signature → `_build_header/footer` |
| Schedule Meeting | `POST /api/email/schedule-meeting` | `handleScheduleMeeting()` | Supabase `meeting_proposals` |
