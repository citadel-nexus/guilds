```python
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter
from opentelemetry.sdk.resources import Resource
from functools import wraps
import time

# Initialize tracer
resource = Resource(attributes={
    "service.name": "citadel-nexus-cognition",
    "service.version": "1.0.0",
    "deployment.environment": "production"
})

trace.set_tracer_provider(TracerProvider(resource=resource))
tracer = trace.get_tracer(__name__)

# Configure OTLP exporter
otlp_exporter = OTLPSpanExporter(
    endpoint="http://localhost:4317",  # SigNoz gRPC endpoint
    insecure=True
)
trace.get_tracer_provider().add_span_processor(
    BatchSpanProcessor(otlp_exporter)
)

def trace_caps_component(component_name, caps_metadata=None):
    """Decorator to trace CAPS components with metadata."""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            with tracer.start_as_current_span(f"{component_name}.{func.__name__}") as span:
                # Add CAPS metadata as span attributes
                if caps_metadata:
                    span.set_attribute("caps.module", caps_metadata.get('module_name'))
                    span.set_attribute("caps.version", caps_metadata.get('version'))
                    span.set_attribute("caps.quality", caps_metadata.get('quality_score'))
                    span.set_attribute("caps.trust_level", caps_metadata.get('trust_level'))
                    span.set_attribute("caps.recovery_group", caps_metadata.get('recovery_group'))
                
                # Execute function
                start_time = time.perf_counter()
                try:
                    result = func(*args, **kwargs)
                    span.set_attribute("success", True)
                    return result
                except Exception as e:
                    span.set_attribute("success", False)
                    span.set_attribute("error.type", type(e).__name__)
                    span.set_attribute("error.message", str(e))
                    span.record_exception(e)
                    raise
                finally:
                    elapsed_ms = (time.perf_counter() - start_time) * 1000
                    span.set_attribute("latency_ms", elapsed_ms)
        return wrapper
    return decorator
```

```python
from otel_tracer import trace_caps_component, tracer

class CAPSProtocol:
    def __init__(self):
        self.metadata = {
            'module_name': 'caps_protocol',
            'version': '1.0.0',
            'quality_score': 0.95,
            'trust_level': 'PRODUCTION',
            'recovery_group': 'CORE'
        }
    
    @trace_caps_component("CAPS", caps_metadata={'module_name': 'caps_protocol'})
    def process_query(self, query):
        with tracer.start_as_current_span("caps.header_validation") as span:
            span.set_attribute("query.length", len(query))
            # Validate and attach CAPS header
            header = self._build_header(query)
        
        with tracer.start_as_current_span("caps.ledger_write"):
            # Log to CAPS ledger
            self._write_ledger(query, header)
        
        return header
```

## 🧬 Phase 6: Gephi Export & Analysis (Week 3)


### Export Traces to CSV

**Script**: `export_signoz_``traces.py`
```python
import requests
import pandas as pd
from datetime import datetime, timedelta

def export_traces_for_gephi(signoz_url, days=7):
    """Export SigNoz traces as Gephi-compatible CSV."""
    
    # Query SigNoz API
    end_time = datetime.now()
    start_time = end_time - timedelta(days=days)
    
    traces = requests.post(f"{signoz_url}/api/v1/traces", json={
        "start": int(start_time.timestamp() * 1000),
        "end": int(end_time.timestamp() * 1000),
        "limit": 10000
    }).json()
    
    # Build node list
    nodes = {}
    edges = []
    
    for trace in traces['data']:
        for span in trace['spans']:
            service = span.get('serviceName')
            operation = span.get('operationName')
            
            # Extract component from operation (e.g., "CAPS.process_query")
            component = operation.split('.')[0] if '.' in operation else service
            
            # Add node
            if component not in nodes:
                nodes[component] = {
                    'Id': component,
                    'Label': component,
                    'AvgLatency': [],
                    'ErrorCount': 0,
                    'CallCount': 0
                }
            
            # Update metrics
            nodes[component]['CallCount'] += 1
            nodes[component]['AvgLatency'].append(span.get('duration', 0) / 1000)  # ms
            
            if span.get('statusCode') == 2:  # Error
                nodes[component]['ErrorCount'] += 1
            
            # Build edges from parent-child relationships
            if 'references' in span:
                for ref in span['references']:
                    if ref.get('refType') == 'CHILD_OF':
                        parent_span = next(
                            (s for s in trace['spans'] if s['spanID'] == ref['spanID']),
                            None
                        )
                        if parent_span:
                            parent_component = parent_span['operationName'].split('.')[0]
                            edges.append({
                                'Source': parent_component,
                                'Target': component,
                                'Weight': 1,
                                'Latency': span.get('duration', 0) / 1000
                            })
    
    # Calculate node metrics
    nodes_df = pd.DataFrame([
        {
            'Id': node['Id'],
            'Label': node['Label'],
            'AvgLatency': sum(node['AvgLatency']) / len(node['AvgLatency']) if node['AvgLatency'] else 0,
            'ErrorCount': node['ErrorCount'],
            'CallCount': node['CallCount'],
            'ErrorRate': node['ErrorCount'] / node['CallCount'] if node['CallCount'] > 0 else 0
        }
        for node in nodes.values()
    ])
    
    # Aggregate edges
    edges_df = pd.DataFrame(edges)
    edges_agg = edges_df.groupby(['Source', 'Target']).agg({
        'Weight': 'sum',
        'Latency': 'mean'
    }).reset_index()
    
    # Save CSVs
    nodes_df.to_csv('gephi_nodes.csv', index=False)
    edges_agg.to_csv('gephi_edges.csv', index=False)
    
    print(f"Exported {len(nodes_df)} nodes and {len(edges_agg)} edges")
    return nodes_df, edges_agg

# Run export
export_traces_for_gephi("http://localhost:3301", days=7)
```

## 🧠 AI Cognition Map: Cause → Effect → Data Gain


### Query Processing Chain (Causal Flow)

**Stage 1: Input Reception**
- **Cause**: User query arrives
- **Effect**: CAPS Protocol activated
- **Data Gain**: Query metadata (length, timestamp, source, context ID)
- **Instrumentation**: `caps.header_validation` span
↓
**Stage 2: Classification**
- **Cause**: Query + CAPS header
- **Effect**: Enumeration classifier detects intent categories
- **Data Gain**: Primary category, secondary category, confidence score (0-1)
- **Instrumentation**: `enum.category_detection` span
- **Observable**: Confidence < 0.6 triggers uncertainty flag
↓
**Stage 3: Context Building**
- **Cause**: Classified query + historical context
- **Effect**: Knowledge graph constructed
- **Data Gain**: Entity list, relationship map, semantic embeddings (768-dim vectors)
- **Instrumentation**: `context.entity_extraction`, `context.embedding`, `context.graph_build` spans
- **Observable**: Entity count, graph depth, embedding latency
↓
**Stage 4: Memory Retrieval**
- **Cause**: Query embeddings
- **Effect**: FAISS vector search
- **Data Gain**: Top-K similar past interactions (K=5-20), similarity scores
- **Instrumentation**: `faiss.search` span
- **Observable**: Search latency (<50ms target), recall quality
↓
**Stage 5: Response Generation**
- **Cause**: Context graph + retrieved memories + knowledge base
- **Effect**: Response synthesized
- **Data Gain**: Response text, confidence score, cited sources, token count
- **Instrumentation**: `response.synthesis` span
- **Observable**: Generation time, token/sec rate
↓
**Stage 6: Belief Update**
- **Cause**: New interaction completed
- **Effect**: Belief system updated with learned patterns
- **Data Gain**: Updated belief scores, new pattern entries, reinforced connections
- **Instrumentation**: `belief.update` span
↓
**Stage 7: Memory Storage**
- **Cause**: Interaction complete
- **Effect**: ULMS writes to SQLite + FAISS index update
- **Data Gain**: Persistent interaction record, searchable vector, rehydration metadata
- **Instrumentation**: `ulms.memory_write`, `ulms.faiss_update` spans

- FAISS search >100ms → Index size issue, recommend partitioning

### Prompt Engineering Insights from Traces

**Use Case**: Optimize prompts by analyzing trace data
**Analysis 1: High Latency Queries**
```sql
-- Find queries with >1s total latency
SELECT query_text, SUM(span_duration) as total_ms
FROM signoz_traces
WHERE service_name = 'citadel-nexus-cognition'
GROUP BY trace_id, query_text
HAVING total_ms > 1000
ORDER BY total_ms DESC
LIMIT 20;
```
**Insight**: Queries with >10 entities in context graph cause 3x slowdown → **Recommendation**: Add entity limit or prompt users to be more specific
**Analysis 2: Low Confidence Classifications**
```sql
-- Find queries where enum confidence < 0.6
SELECT query_text, enum_confidence, response_quality
FROM signoz_traces
WHERE span_name = 'enum.category_detection'
  AND enum_confidence < 0.6;
```
**Insight**: Ambiguous queries have low confidence → **Recommendation**: Prompt clarification questions before proceeding
**Analysis 3: FAISS Recall Quality**
```sql
-- Measure if retrieved memories were actually used in response
SELECT 
  query_id,
  COUNT(retrieved_memories) as fetched,
  COUNT(cited_sources) as used,
  (used / fetched) as utilization_rate
FROM signoz_traces
WHERE span_name IN ('faiss.search', 'response.synthesis')
GROUP BY query_id
HAVING utilization_rate < 0.3;
```
**Insight**: Low utilization means FAISS retrieving irrelevant memories → **Recommendation**: Tune embedding model or adjust Top-K parameter

```python
def export_component_density(days=7):
    """Export component activity density for heatmap overlay."""

    start_time = end_time - (days * 86400)
    
    components = ['CAPS', 'Enum', 'Context', 'FAISS', 'Response', 'Belief', 'ULMS']

            start=start_time,
            end=end_time,
            query=query

                'component': component,
                'density_score': total_count,

    # Normalize to 0-1 scale
    df = pd.DataFrame(density_data)
    max_density = df['density_score'].max()
    df['normalized_density'] = df['density_score'] / max_density
    
    # Export for Gephi
    df.to_csv('gephi_density_overlay.csv', index=False)
    return df
```

```python
import requests
import pandas as pd

POSTHOG_API_KEY = '<your-project-key>'
POSTHOG_HOST = 'https://app.posthog.com'

def export_edge_heatmap(days=7):
    """Export edge traversal frequency for heatmap."""
    
    # Query PostHog events API
    url = f'{POSTHOG_HOST}/api/projects/<project-id>/events/'
    headers = {'Authorization': f'Bearer {POSTHOG_API_KEY}'}
    
    params = {
        'event': 'cognition_edge_traversal',
        'after': f'-{days}d',
    }
    
    response = requests.get(url, headers=headers, params=params)
    events = response.json()['results']
    
    # Aggregate edge traversals
    edge_counts = {}
    for event in events:
        edge_id = event['properties']['edge_id']
        edge_counts[edge_id] = edge_counts.get(edge_id, 0) + 1
    
    # Convert to dataframe
    df = pd.DataFrame([
        {
            'source': edge.split('->')[0],
            'target': edge.split('->')[1],
            'traversal_count': count,
            'heatmap_intensity': 0  # Will normalize
        }
        for edge, count in edge_counts.items()
    ])
    
    # Normalize to 0-1
    max_count = df['traversal_count'].max()
    df['heatmap_intensity'] = df['traversal_count'] / max_count
    
    # Export
    df.to_csv('gephi_edge_heatmap.csv', index=False)
    return df
```

### How the System Works: Architecture Overview

**Data Flow Pipeline**:
1. **Application Layer**: OpenTelemetry instruments all components (CAPS, Enum, Context, etc.)
1. **Multi-Export Bridge**: Traces/metrics sent to SigNoz + DataDog simultaneously
1. **Behavioral Tracking**: PostHog captures user pathways and interaction patterns
1. **Data Aggregation**: Weekly export scripts pull from all three sources
1. **Visualization Engine**: Gephi merges data layers and renders density-based cognition map
1. **Artifact Storage**: PNG exports archived with metadata for historical analysis
**Key Concepts**:
- **Cognitive Mass**: Activity density determines node size (high activity = large mass)
- **Thermal Mapping**: Heatmap gradient shows hot zones (blue = cold, red = hot)
- **Gravitational Clustering**: Force Atlas pulls high-mass nodes to center
- **Edge Intensity**: Pathway traversal frequency determines edge thickness/color

```python
def trace_caps_component(component_name, caps_metadata=None):
    """
    Decorator to trace CAPS-aware components with automatic:
    - OpenTelemetry span creation
    - DataDog metric emission
    - CAPS metadata attachment
    - Error tracking
    """
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # Start OpenTelemetry span
            with tracer.start_as_current_span(f"{component_name.lower()}.execute") as span:
                # Attach CAPS metadata
                if caps_metadata:
                    for key, value in caps_metadata.items():
                        span.set_attribute(f"caps.{key}", value)
                
                span.set_attribute("component.name", component_name)
                
                # Track start time for latency
                start_time = time.time()
                
                # Increment DataDog counter
                statsd.increment(
                    f'cognition.{component_name.lower()}.invocation',
                    tags=[f'component:{component_name}']
                )
                
                # Track active load (density metric)
                statsd.increment(f'cognition.{component_name.lower()}.density')
                
                try:
                    # Execute function
                    result = func(*args, **kwargs)
                    
                    # Record success
                    span.set_attribute("success", True)
                    statsd.increment(
                        f'cognition.{component_name.lower()}.success',
                        tags=[f'component:{component_name}']
                    )
                    
                    return result
                    
                except Exception as e:
                    # Record error
                    span.set_attribute("success", False)
                    span.set_attribute("error.type", type(e).__name__)
                    span.set_attribute("error.message", str(e))
                    span.record_exception(e)
                    
                    statsd.increment(
                        f'cognition.{component_name.lower()}.error',
                        tags=[
                            f'component:{component_name}',
                            f'error_type:{type(e).__name__}'
                        ]
                    )
                    
                    raise
                    
                finally:
                    # Record latency
                    duration_ms = (time.time() - start_time) * 1000
                    span.set_attribute("duration_ms", duration_ms)
                    
                    statsd.histogram(
                        f'cognition.{component_name.lower()}.duration',
                        duration_ms,
                        tags=[f'component:{component_name}']
                    )
                    
                    # Decrement density
                    statsd.decrement(f'cognition.{component_name.lower()}.density')
        
        return wrapper
    return decorator


class CognitionLoadTracker:
    """
    Tracks system-wide cognitive load for mass visualization.
    Maintains gauge of active queries across all components.
    """
    def __init__(self):
        self.active_queries = 0
        self.component_loads = {}
    
    def start_query(self, component, query_id):
        """Track query start for density calculation."""
        self.active_queries += 1
        self.component_loads[component] = self.component_loads.get(component, 0) + 1
        
        # Update DataDog gauges
        statsd.gauge('cognition.active_queries.total', self.active_queries)
        statsd.gauge(
            f'cognition.active_queries.by_component',
            self.component_loads[component],
            tags=[f'component:{component}']
        )
        
        # Emit event for time-series analysis
        statsd.increment(
            'cognition.query.start',
            tags=[
                f'component:{component}',
                f'query_id:{query_id}'
            ]
        )
    
    def end_query(self, component, query_id):
        """Track query end for density calculation."""
        self.active_queries = max(0, self.active_queries - 1)
        self.component_loads[component] = max(0, self.component_loads.get(component, 0) - 1)
        
        # Update DataDog gauges
        statsd.gauge('cognition.active_queries.total', self.active_queries)
        statsd.gauge(
            f'cognition.active_queries.by_component',
            self.component_loads[component],
            tags=[f'component:{component}']
        )
        
        # Emit event
        statsd.increment(
            'cognition.query.end',
            tags=[
                f'component:{component}',
                f'query_id:{query_id}'
            ]
        )
    
    def get_load_distribution(self):
        """Get current load distribution across components."""
        return {
            'total': self.active_queries,
            'by_component': dict(self.component_loads)
        }
```

```python
"""
PostHog integration for tracking user pathways and interaction patterns.
Creates heatmap data for edge traversal visualization.
"""

import posthog
from functools import wraps
import time
import os
from typing import List

# Initialize PostHog
posthog.project_api_key = os.getenv("POSTHOG_API_KEY", "")
posthog.host = os.getenv("POSTHOG_HOST", "https://app.posthog.com")
posthog.disabled = not posthog.project_api_key  # Disable if no API key

if posthog.project_api_key:
    print("✓ PostHog initialized")
else:
    print("⚠ PostHog disabled (set POSTHOG_API_KEY to enable)")


def track_cognition_event(event_name, component):
    """
    Decorator to track cognition events in PostHog.
    Creates behavioral data for pathway heatmaps.
    """
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if posthog.disabled:
                return func(*args, **kwargs)
            
            event_id = f"{int(time.time() * 1000)}_{component}"
            
            # Capture event start
            posthog.capture(
                distinct_id='citadel-system',
                event=f'cognition_{event_name}_start',
                properties={
                    'component': component,
                    'event_id': event_id,
                    'timestamp': time.time(),
                    '$set': {'last_component': component}
                }
            )
            
            start_time = time.time()
            error_occurred = None
            
            try:
                # Execute function
                result = func(*args, **kwargs)
                
                # Capture event completion
                posthog.capture(
                    distinct_id='citadel-system',
                    event=f'cognition_{event_name}_complete',
                    properties={
                        'component': component,
                        'event_id': event_id,
                        'timestamp': time.time(),
                        'duration_ms': (time.time() - start_time) * 1000,
                        'success': True,
                    }
                )
                
                return result
                
            except Exception as e:
                error_occurred = e
                
                # Capture error
                posthog.capture(
                    distinct_id='citadel-system',
                    event=f'cognition_{event_name}_error',
                    properties={
                        'component': component,
                        'event_id': event_id,
                        'timestamp': time.time(),
                        'duration_ms': (time.time() - start_time) * 1000,
                        'success': False,
                        'error_type': type(e).__name__,
                    }
                )
                
                raise
            
        return wrapper
    return decorator


class CognitionPathTracker:
    """
    Tracks complete pathways through the cognition graph.
    Generates edge traversal heatmap data.
    """
    def __init__(self):
        self.active_pathways = {}
    
    def start_pathway(self, query_id: str, initial_component: str):
        """Initialize a new pathway track."""
        self.active_pathways[query_id] = {
            'components': [initial_component],
            'start_time': time.time(),
            'query_id': query_id,
        }
        
        if not posthog.disabled:
            posthog.capture(
                distinct_id='citadel-system',
                event='pathway_start',
                properties={
                    'query_id': query_id,
                    'initial_component': initial_component,
                    'timestamp': time.time(),
                }
            )
    
    def add_component(self, query_id: str, component: str):
        """Add a component to the pathway."""
        if query_id not in self.active_pathways:
            # Pathway not tracked, start it
            self.start_pathway(query_id, component)
            return
        
        pathway = self.active_pathways[query_id]
        pathway['components'].append(component)
        
        # Track edge traversal
        if len(pathway['components']) > 1:
            source = pathway['components'][-2]
            target = component
            edge_id = f"{source}->{target}"
            
            if not posthog.disabled:
                posthog.capture(
                    distinct_id='citadel-system',
                    event='cognition_edge_traversal',
                    properties={
                        'query_id': query_id,
                        'source': source,
                        'target': target,
                        'edge_id': edge_id,
                        'timestamp': time.time(),
                    }
                )
    
    def complete_pathway(self, query_id: str):
        """Mark pathway as complete and emit summary event."""
        if query_id not in self.active_pathways:
            return
        
        pathway = self.active_pathways.pop(query_id)
        pathway_sequence = pathway['components']
        pathway_hash = "->".join(pathway_sequence)
        duration_ms = (time.time() - pathway['start_time']) * 1000
        
        if not posthog.disabled:
            posthog.capture(
                distinct_id='citadel-system',
                event='cognition_pathway',
                properties={
                    'query_id': query_id,
                    'pathway': pathway_hash,
                    'length': len(pathway_sequence),
                    'components': pathway_sequence,
                    'duration_ms': duration_ms,
                    'timestamp': time.time(),
                }
            )
    
    def track_classification_result(self, query: str, result: dict):
        """Track classification results for heatmap density."""
        if posthog.disabled:
            return
        
        posthog.capture(
            distinct_id='citadel-system',
            event='classification_result',
            properties={
                'primary_category': result.get('primary'),
                'secondary_category': result.get('secondary'),
                'confidence': result.get('confidence'),
                'query_length': len(query),
                'timestamp': time.time(),
            }
        )


# Global pathway tracker instance
pathway_tracker = CognitionPathTracker()
```

```python
            'trust_level': 'PRODUCTION',

        """

            # Validate and attach CAPS header

                span.set_attribute("query.id", query_id)

                'query_id': query_id,
                'header': header,
                'query': query,
```

```python
"""
Export DataDog density metrics to CSV for Gephi overlay.
Normalizes activity density across all components.
"""

from datadog import initialize, api
import pandas as pd
import numpy as np
import time
import os
import argparse

# Initialize DataDog
options = {
    'api_key': os.getenv('DD_API_KEY'),
    'app_key': os.getenv('DD_APP_KEY')
}

initialize(**options)

COMPONENTS = ['CAPS', 'Enum', 'Context', 'FAISS', 'Response', 'Belief', 'ULMS', 'MCA', 'TaskIR', 'Reflex']


def export_component_density(days=7, output_file='gephi_density_overlay.csv'):
    """
    Export component activity density for heatmap overlay.
    
    Returns CSV with columns:
    - component: Component name
    - density_score: Raw activity count
    - normalized_density: 0-1 normalized score for heatmap coloring
    - invocation_count: Total invocations
    - error_rate: Error percentage
    - avg_duration_ms: Average processing time
    """
    print(f"📊 Exporting DataDog density metrics ({days} days)...")
    
    end_time = int(time.time())
    start_time = end_time - (days * 86400)
    
    density_data = []
    
    for component in COMPONENTS:
        print(f"  Querying {component}...")
        
        try:
            # Query density metric
            density_query = f'sum:cognition.{component.lower()}.density*.as_count()'
            density_result = api.Metric.query(
                start=start_time,
                end=end_time,
                query=density_query
            )
            
            # Query invocation count
            invocation_query = f'sum:cognition.{component.lower()}.invocation*.as_count()'
            invocation_result = api.Metric.query(
                start=start_time,
                end=end_time,
                query=invocation_query
            )
            
            # Query error count
            error_query = f'sum:cognition.{component.lower()}.error*.as_count()'
            error_result = api.Metric.query(
                start=start_time,
                end=end_time,
                query=error_query
            )
            
            # Query duration
            duration_query = f'avg:cognition.{component.lower()}.duration*'
            duration_result = api.Metric.query(
                start=start_time,
                end=end_time,
                query=duration_query
            )
            
            # Extract values
            density_score = 0
            if density_result.get('series'):
                density_score = sum([point[1] for point in density_result['series'][0]['pointlist']])
            
            invocation_count = 0
            if invocation_result.get('series'):
                invocation_count = sum([point[1] for point in invocation_result['series'][0]['pointlist']])
            
            error_count = 0
            if error_result.get('series'):
                error_count = sum([point[1] for point in error_result['series'][0]['pointlist']])
            
            avg_duration_ms = 0
            if duration_result.get('series'):
                durations = [point[1] for point in duration_result['series'][0]['pointlist']]
                avg_duration_ms = np.mean(durations) if durations else 0
            
            # Calculate error rate
            error_rate = (error_count / invocation_count * 100) if invocation_count > 0 else 0
            
            density_data.append({
                'component': component,
                'density_score': density_score,
                'normalized_density': 0,  # Will normalize below
                'invocation_count': invocation_count,
                'error_rate': round(error_rate, 2),
                'avg_duration_ms': round(avg_duration_ms, 2),
            })
            
            print(f"    ✓ {component}: {int(density_score)} density, {int(invocation_count)} invocations")
            
        except Exception as e:
            print(f"    ⚠ Error querying {component}: {e}")
            # Add placeholder
            density_data.append({
                'component': component,
                'density_score': 0,
                'normalized_density': 0,
                'invocation_count': 0,
                'error_rate': 0,
                'avg_duration_ms': 0,
            })
    
    # Convert to DataFrame
    df = pd.DataFrame(density_data)
    
    # Normalize density to 0-1 scale
    max_density = df['density_score'].max()
    if max_density > 0:
        df['normalized_density'] = df['density_score'] / max_density
    else:
        df['normalized_density'] = 0
    
    # Round for readability
    df['normalized_density'] = df['normalized_density'].round(3)
    
    # Export
    df.to_csv(output_file, index=False)
    print(f"\n✅ Exported to {output_file}")
    print(f"\nDensity Summary:")
    print(df.to_string(index=False))
    
    return df


if __name__ == '__main__':
    parser = argparse.ArgumentParser(description='Export DataDog density metrics')
    parser.add_argument('--days', type=int, default=7, help='Number of days to query')
    parser.add_argument('--output', type=str, default='gephi_density_overlay.csv', help='Output filename')
    
    args = parser.parse_args()
    export_component_density(days=args.days, output_file=args.output)
```

```python
"""
Export PostHog edge traversal heatmap to CSV for Gephi.
Creates edge intensity overlay based on pathway usage.
"""

import requests
import pandas as pd
import numpy as np
import time
import os
import argparse
from datetime import datetime, timedelta

POSTHOG_API_KEY = os.getenv('POSTHOG_API_KEY')
POSTHOG_HOST = os.getenv('POSTHOG_HOST', 'https://app.posthog.com')
PROJECT_ID = os.getenv('POSTHOG_PROJECT_ID')


def export_edge_heatmap(days=7, output_file='gephi_edge_heatmap.csv'):
    """
    Export edge traversal frequency for heatmap.
    
    Returns CSV with columns:
    - source: Source component
    - target: Target component
    - traversal_count: Number of times edge was traversed
    - heatmap_intensity: 0-1 normalized intensity for coloring
    - avg_duration_ms: Average time between source and target
    """
    print(f"🔥 Exporting PostHog edge heatmap ({days} days)...")
    
    if not POSTHOG_API_KEY or not PROJECT_ID:
        print("⚠ PostHog not configured. Set POSTHOG_API_KEY and POSTHOG_PROJECT_ID")
        return pd.DataFrame()
    
    # Calculate time range
    end_date = datetime.now()
    start_date = end_date - timedelta(days=days)
    
    # Query PostHog events API
    url = f'{POSTHOG_HOST}/api/projects/{PROJECT_ID}/events/'
    headers = {'Authorization': f'Bearer {POSTHOG_API_KEY}'}
    
    params = {
        'event': 'cognition_edge_traversal',
        'after': start_date.isoformat(),
        'before': end_date.isoformat(),
    }
    
    try:
        print(f"  Querying PostHog API...")
        response = requests.get(url, headers=headers, params=params, timeout=30)
        response.raise_for_status()
        
        events = response.json().get('results', [])
        print(f"  ✓ Retrieved {len(events)} edge traversal events")
        
    except Exception as e:
        print(f"  ⚠ Error querying PostHog: {e}")
        return pd.DataFrame()
    
    if not events:
        print("  ⚠ No events found")
        return pd.DataFrame()
    
    # Aggregate edge traversals
    edge_data = {}
    
    for event in events:
        props = event.get('properties', {})
        edge_id = props.get('edge_id', '')
        
        if not edge_id or '->' not in edge_id:
            continue
        
        source, target = edge_id.split('->', 1)
        
        if edge_id not in edge_data:
            edge_data[edge_id] = {
                'source': source,
                'target': target,
                'traversal_count': 0,
                'timestamps': [],
            }
        
        edge_data[edge_id]['traversal_count'] += 1
        edge_data[edge_id]['timestamps'].append(event.get('timestamp'))
    
    # Convert to DataFrame
    rows = []
    for edge_id, data in edge_data.items():
        rows.append({
            'source': data['source'],
            'target': data['target'],
            'traversal_count': data['traversal_count'],
            'heatmap_intensity': 0,  # Will normalize
        })
    
    df = pd.DataFrame(rows)
    
    if df.empty:
        print("  ⚠ No edge data to export")
        return df
    
    # Normalize to 0-1 for heatmap intensity
    max_count = df['traversal_count'].max()
    if max_count > 0:
        df['heatmap_intensity'] = df['traversal_count'] / max_count
    else:
        df['heatmap_intensity'] = 0
    
    # Round
    df['heatmap_intensity'] = df['heatmap_intensity'].round(3)
    
    # Sort by traversal count
    df = df.sort_values('traversal_count', ascending=False)
    
    # Export
    df.to_csv(output_file, index=False)
    print(f"\n✅ Exported to {output_file}")
    print(f"\nTop 10 Hottest Edges:")
    print(df.head(10).to_string(index=False))
    
    return df


if __name__ == '__main__':
    parser = argparse.ArgumentParser(description='Export PostHog edge heatmap')
    parser.add_argument('--days', type=int, default=7, help='Number of days to query')
    parser.add_argument('--output', type=str, default='gephi_edge_heatmap.csv', help='Output filename')
    
    args = parser.parse_args()
    export_edge_heatmap(days=args.days, output_file=args.output)
```

**1. Application Runtime (Instrumented Components)**
```javascript

   ├─ @track_cognition_event("process")    → Creates PostHog event
   ├─ load_tracker.start_query()            → Increments density counter
   └─ pathway_tracker.start_pathway()       → Begins pathway track
   ↓

   ├─ @trace_caps_component("Enum")        → New span (child of CAPS span)
   ├─ @track_cognition_event("classify")   → PostHog event
   ├─ pathway_tracker.add_component()       → Records CAPS→Enum edge
   └─ Emits classification_result event     → PostHog heatmap data
   ↓
... (Context, FAISS, Response, etc.)
   ↓
pathway_tracker.complete_pathway()         → Emits full pathway event
load_tracker.end_query()                   → Decrements density counter
```
**2. Data Collection (Multi-Platform Storage)**
```javascript
OpenTelemetry Spans  →  SigNoz (base graph structure)
DataDog Metrics      →  DataDog (density/mass data)
PostHog Events       →  PostHog (pathway/heatmap data)
```

## 🎮 Phase 8: XP/TP Performance System — Gamified Cognition Metrics


### Overview: Performance-Driven Density

**Concept**: Components earn **XP** (Experience Points) and **TP** (Training Points) based on performance metrics. Good performance increases cognitive "mass" in visualizations, poor performance reduces it.
**Core Mechanics**:
- **XP** = Long-term component reliability score (0-10,000 scale)
- **TP** = Short-term training effectiveness (resets weekly, 0-1,000 scale)
- **Performance Metrics** → XP/TP changes → Density multiplier → Gephi visualization size/color
**Visual Impact**:
- High XP/TP → Larger, brighter nodes (healthy, well-trained components)
- Low XP/TP → Smaller, dimmer nodes (struggling components need attention)
- XP/TP decline → Visual "decay" effect showing degradation over time

---


### Performance Metrics That Affect XP/TP

| **Metric** | **XP Gain** | **TP Gain** | **Condition** |
| Successful query completion | +10 | +5 | P95 latency < target |
| High confidence classification | +15 | +8 | Enum confidence > 0.85 |
| Fast response | +20 | +10 | Total latency < 500ms |
| User feedback positive | +50 | +25 | Explicit positive feedback |
| Zero errors in 1 hour | +30 | +15 | Error rate = 0% |
| FAISS recall success | +12 | +6 | Retrieved memory used in response |
| Reflex auto-recovery | +40 | +20 | Successful automatic recovery |
| **Metric** | **XP Loss** | **TP Loss** | **Condition** |
| Query failure | -25 | -15 | Exception raised |
| Timeout | -30 | -20 | Latency > 3x target |
| Low confidence result | -10 | -8 | Enum confidence < 0.5 |
| User feedback negative | -100 | -50 | Explicit negative feedback |
| High error rate | -20/hour | -10/hour | Error rate > 5% |
| FAISS recall failure | -15 | -10 | Retrieved memory not used |
| Integration failure | -40 | -25 | External service unavailable |
- **XP Decay**: -1 per day of inactivity (encourages continuous use)
- **TP Decay**: -10% per day (short-term training requires consistency)
- **Minimum**: XP cannot drop below 100 (baseline competence)

### XP/TP Calculation Engine

**File**: `citadel-helper/performance/xp_tp_`engine.py
```python
"""
XP/TP calculation engine with Supabase integration.
Tracks component performance and adjusts density metrics.
"""

from supabase import create_client, Client
import os
from datetime import datetime, timedelta
from typing import Dict, Optional
import math

class XPTPEngine:
    def __init__(self):
        self.supabase: Client = create_client(
            os.getenv("SUPABASE_URL"),
            os.getenv("SUPABASE_KEY")
        )
        
        # XP/TP configuration
        self.XP_GAINS = {
            'query_success': 10,
            'high_confidence': 15,
            'fast_response': 20,
            'positive_feedback': 50,
            'zero_errors_1h': 30,
            'faiss_recall_success': 12,
            'auto_recovery': 40,
        }
        
        self.XP_LOSSES = {
            'query_failure': -25,
            'timeout': -30,
            'low_confidence': -10,
            'negative_feedback': -100,
            'high_error_rate': -20,
            'faiss_recall_failure': -15,
            'integration_failure': -40,
        }
        
        self.TP_MULTIPLIER = 0.5  # TP gains/losses are 50% of XP
        self.XP_DECAY_DAILY = -1
        self.TP_DECAY_DAILY_PCT = 0.10
        self.MIN_XP = 100
    
    def record_performance_event(
        self,
        component_name: str,
        event_type: str,
        query_id: str,
        trace_id: str,
        latency_ms: Optional[int] = None,
        confidence_score: Optional[float] = None,
        error_message: Optional[str] = None,
        metadata: Optional[Dict] = None
    ):
        """Record a performance event and update XP/TP."""
        
        # Calculate XP/TP changes
        xp_change = self.XP_GAINS.get(event_type) or self.XP_LOSSES.get(event_type, 0)
        tp_change = int(xp_change * self.TP_MULTIPLIER)
        
        # Insert event
        self.supabase.table('performance_events').insert({
            'component_name': component_name,
            'event_type': event_type,
            'xp_change': xp_change,
            'tp_change': tp_change,
            'query_id': query_id,
            'trace_id': trace_id,
            'latency_ms': latency_ms,
            'confidence_score': confidence_score,
            'error_message': error_message,
            'metadata': metadata or {}
        }).execute()
        
        # Update component performance
        self._update_component_performance(component_name, xp_change, tp_change, event_type)
    
    def _update_component_performance(
        self,
        component_name: str,
        xp_change: int,
        tp_change: int,
        event_type: str
    ):
        """Update component XP/TP and stats."""
        
        # Get current performance
        result = self.supabase.table('component_performance').select('*').eq(
            'component_name', component_name
        ).execute()
        
        if not result.data:
            # Initialize new component
            self.supabase.table('component_performance').insert({
                'component_name': component_name,
                'xp': max(1000 + xp_change, self.MIN_XP),
                'tp': max(500 + tp_change, 0),
                'total_invocations': 1,
                'successful_invocations': 1 if xp_change > 0 else 0,
                'failed_invocations': 0 if xp_change > 0 else 1,
                'last_activity_at': datetime.utcnow().isoformat(),
            }).execute()
        else:
            current = result.data[0]
            new_xp = max(current['xp'] + xp_change, self.MIN_XP)
            new_tp = max(current['tp'] + tp_change, 0)
            
            # Update
            self.supabase.table('component_performance').update({
                'xp': new_xp,
                'tp': new_tp,
                'total_invocations': current['total_invocations'] + 1,
                'successful_invocations': current['successful_invocations'] + (1 if xp_change > 0 else 0),
                'failed_invocations': current['failed_invocations'] + (0 if xp_change > 0 else 1),
                'last_activity_at': datetime.utcnow().isoformat(),
                'updated_at': datetime.utcnow().isoformat(),
            }).eq('component_name', component_name).execute()
    
    def apply_daily_decay(self):
        """Apply daily XP/TP decay to all components."""
        
        result = self.supabase.table('component_performance').select('*').execute()
        
        for component in result.data:
            # Check inactivity
            last_activity = datetime.fromisoformat(component['last_activity_at'])
            days_inactive = (datetime.utcnow() - last_activity).days
            
            if days_inactive > 0:
                # Apply XP decay
                xp_loss = self.XP_DECAY_DAILY * days_inactive
                new_xp = max(component['xp'] + xp_loss, self.MIN_XP)
                
                # Apply TP decay (percentage-based)
                tp_decay = component['tp'] * (self.TP_DECAY_DAILY_PCT * days_inactive)
                new_tp = max(component['tp'] - tp_decay, 0)
                
                self.supabase.table('component_performance').update({
                    'xp': new_xp,
                    'tp': int(new_tp),
                    'updated_at': datetime.utcnow().isoformat(),
                }).eq('component_name', component['component_name']).execute()
    
    def calculate_density_multiplier(self, component_name: str) -> float:
        """
        Calculate density multiplier based on XP/TP.
        Returns value between 0.5 (poor performance) and 2.0 (excellent performance).
        """
        
        result = self.supabase.table('component_performance').select('*').eq(
            'component_name', component_name
        ).execute()
        
        if not result.data:
            return 1.0  # Neutral multiplier for new components
        
        component = result.data[0]
        xp = component['xp']
        tp = component['tp']
        
        # XP contribution (0.5 to 1.5 multiplier)
        # Base XP = 1000, scales logarithmically
        xp_factor = 0.5 + (math.log10(xp / 100) / 2)
        xp_factor = max(0.5, min(1.5, xp_factor))
        
        # TP contribution (0.0 to 0.5 boost)
        # Max TP = 1000
        tp_boost = (tp / 1000) * 0.5
        
        # Combined multiplier
        multiplier = xp_factor + tp_boost
        return round(multiplier, 2)
    
    def get_leaderboard(self, limit: int = 10):
        """Get top components by XP."""
        
        result = self.supabase.table('component_performance').select('*').order(
            'xp', desc=True
        ).limit(limit).execute()
        
        return result.data
    
    def export_density_snapshot(self):
        """Export current density snapshot for Gephi."""
        
        result = self.supabase.table('component_performance').select('*').execute()
        
        snapshots = []
        for component in result.data:
            multiplier = self.calculate_density_multiplier(component['component_name'])
            
            # Calculate adjusted density (assumes base density from DataDog)
            # This would be combined with DataDog metrics in production
            base_density = component['total_invocations']
            adjusted_density = base_density * multiplier
            
            # Normalize for Gephi (0-1 scale)
            # This is simplified; real implementation would normalize across all components
            max_density = 10000
            normalized = min(adjusted_density / max_density, 1.0)
            
            # Color coding based on performance
            if multiplier >= 1.5:
                color = "green"
            elif multiplier >= 1.0:
                color = "blue"
            elif multiplier >= 0.75:
                color = "yellow"
            else:
                color = "red"
            
            snapshot = {
                'snapshot_date': datetime.utcnow().isoformat(),
                'component_name': component['component_name'],
                'raw_density': base_density,
                'xp_multiplier': multiplier,
                'tp_multiplier': component['tp'] / 1000,
                'adjusted_density': adjusted_density,
                'normalized_density': normalized,
                'node_size': int(50 + (normalized * 150)),  # 50-200 range
                'node_color': color,
                'halo_intensity': normalized,
            }
            
            snapshots.append(snapshot)
        
        # Insert snapshots
        if snapshots:
            self.supabase.table('density_snapshots').insert(snapshots).execute()
        
        return snapshots


# Initialize global engine
xp_tp_engine = XPTPEngine()
```

```python
def trace_caps_component(component_name, caps_metadata=None):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            with tracer.start_as_current_span(
                f"{component_name.lower()}.{func.__name__}",
                kind=trace.SpanKind.INTERNAL
            ) as span:
                # Existing instrumentation...
                start_time = time.time()
                query_id = str(uuid.uuid4())
                trace_id = format(span.get_span_context().trace_id, '032x')
                
                try:
                    result = func(*args, **kwargs)
                    duration_ms = int((time.time() - start_time) * 1000)
                    
                    # Determine performance event type
                    if duration_ms < 500:
                        event_type = 'fast_response'
                    else:
                        event_type = 'query_success'
                    
                    # Record XP/TP event
                    xp_tp_engine.record_performance_event(
                        component_name=component_name,
                        event_type=event_type,
                        query_id=query_id,
                        trace_id=trace_id,
                        latency_ms=duration_ms,
                        metadata={'function': func.__name__}
                    )
                    
                    return result
                    
                except Exception as e:
                    duration_ms = int((time.time() - start_time) * 1000)
                    
                    # Record failure event
                    event_type = 'timeout' if duration_ms > 3000 else 'query_failure'
                    
                    xp_tp_engine.record_performance_event(
                        component_name=component_name,
                        event_type=event_type,
                        query_id=query_id,
                        trace_id=trace_id,
                        latency_ms=duration_ms,
                        error_message=str(e),
                        metadata={'function': func.__name__, 'error_type': type(e).__name__}
                    )
                    
                    raise
        
        return wrapper
    return decorator
```

```python
def calculate_failure_risk(component):
    # DataDog: Error rate trend
    error_trend = dd_api.get_metric_trend(f"error.rate.{component}")
    
    # PostHog: User frustration signals (rage clicks, error events)
    frustration = posthog_api.count_events("user_frustration", {"component": component})
    
    # SigNoz: Trace error spike
    trace_errors = signoz_api.count_errors(component)
    
    # Supabase: XP decay rate
    xp_decay = supabase.query(f"SELECT xp - LAG(xp) OVER (ORDER BY updated_at) FROM component_performance WHERE component_name='{component}'")
    
    # Composite risk score (0-100)
    risk_score = (
        error_trend * 0.3 +
        frustration * 0.2 +
        trace_errors * 0.25 +
        abs(xp_decay) * 0.25
    ) * 100
    
    return risk_score
```

```python
from sklearn.metrics.pairwise import cosine_similarity

def calculate_novelty(response_embedding, historical_embeddings):
```

```sql
CREATE TABLE data_access_metadata (
    table_name TEXT NOT NULL,
    record_id TEXT NOT NULL,
    
    -- Access tracking
    created_at TIMESTAMP DEFAULT NOW(),
    last_accessed_at TIMESTAMP DEFAULT NOW(),
    access_count INTEGER DEFAULT 0,

    current_tier TEXT DEFAULT 'hot', -- hot, warm, cold, archived
    tier_last_changed TIMESTAMP,
    
    -- Size
    record_size_bytes BIGINT,
    
    PRIMARY KEY (table_name, record_id)
```

```sql
-- Find hot-tier data that should be moved to cold storage
SELECT 
    table_name,
    record_id,
    days_since_access,
    record_size_bytes,
    record_size_bytes * 0.02 as monthly_hot_cost,  -- $0.02/GB/month
    record_size_bytes * 0.004 as monthly_cold_cost, -- $0.004/GB/month
    (record_size_bytes * 0.02) - (record_size_bytes * 0.004) as potential_savings
FROM data_access_metadata
WHERE current_tier = 'hot'
  AND days_since_access > 30
  AND record_size_bytes > 1048576  -- > 1MB
ORDER BY potential_savings DESC
LIMIT 100;
```
```sql
CREATE MATERIALIZED VIEW storage_efficiency_report AS
SELECT 
    DATE_TRUNC('day', NOW()) as report_date,
    
    -- By temperature
    SUM(CASE WHEN temperature = 'hot' THEN record_size_bytes ELSE 0 END) / 1e9 as hot_gb,
    SUM(CASE WHEN temperature = 'warm' THEN record_size_bytes ELSE 0 END) / 1e9 as warm_gb,
    SUM(CASE WHEN temperature = 'cold' THEN record_size_bytes ELSE 0 END) / 1e9 as cold_gb,
    
    -- By tier (actual storage location)
    SUM(CASE WHEN current_tier = 'hot' THEN record_size_bytes ELSE 0 END) / 1e9 as tier_hot_gb,
    SUM(CASE WHEN current_tier = 'cold' THEN record_size_bytes ELSE 0 END) / 1e9 as tier_cold_gb,
    
    -- Misalignment (hot-temperature data in hot tier unnecessarily)
    SUM(CASE WHEN temperature = 'cold' AND current_tier = 'hot' 
        THEN record_size_bytes ELSE 0 END) / 1e9 as misaligned_gb,
    
    -- Cost calculation
    SUM(CASE WHEN current_tier = 'hot' THEN record_size_bytes * 0.02 
             WHEN current_tier = 'cold' THEN record_size_bytes * 0.004 
        END) / 1e9 as monthly_cost,
    
    -- Potential savings if optimally tiered
    (SUM(CASE WHEN current_tier = 'hot' THEN record_size_bytes * 0.02 END) -
     SUM(CASE WHEN temperature = 'hot' THEN record_size_bytes * 0.02
              WHEN temperature IN ('warm', 'cold') THEN record_size_bytes * 0.004 END)) / 1e9 
        as potential_monthly_savings
FROM data_access_metadata;

-- Refresh daily
CREATE UNIQUE INDEX ON storage_efficiency_report(report_date);
```

```sql
    COUNT(*) FILTER (WHERE decision_date > NOW() - INTERVAL '90 days') as decisions_last_90_days,
    
    -- Implementation rate
    COUNT(*) FILTER (WHERE status = 'implemented') as implemented_count,
    COUNT(*) FILTER (WHERE status IN ('approved', 'in_progress')) as pending_count,
    COUNT(*) FILTER (WHERE status = 'implemented')::FLOAT / COUNT(*) as implementation_rate,
    
    -- Timeliness
    AVG(EXTRACT(DAY FROM implemented_at - decision_date)) FILTER (WHERE implemented_at IS NOT NULL) 
        as avg_days_to_implement,
    COUNT(*) FILTER (WHERE implementation_deadline < NOW() AND status != 'implemented') 
        as overdue_count,
    
    -- Impact
    AVG(impact_score) FILTER (WHERE impact_score IS NOT NULL) as avg_impact_score,
    AVG(adherence_rate) FILTER (WHERE adherence_rate IS NOT NULL) as avg_adherence_rate,
    
    -- Health score (composite)
    (
        (COUNT(*) FILTER (WHERE status = 'implemented')::FLOAT / NULLIF(COUNT(*), 0)) * 0.4 +
        (1 - COUNT(*) FILTER (WHERE implementation_deadline < NOW() AND status != 'implemented')::FLOAT / NULLIF(COUNT(*), 0)) * 0.3 +
        (AVG(adherence_rate) FILTER (WHERE adherence_rate IS NOT NULL)) * 0.3
```
