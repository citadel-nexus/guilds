### Central Health Monitor

```python
# tools/health_monitor_phase3.py
import httpx, asyncio
from datadog import statsd
import posthog

GUILD_SERVICES = {
    "commerce":      {"url": "http://localhost:8422/api/commerce/health", "critical": True},
    "intelligence":  {"url": "http://localhost:8000/api/intel/health", "critical": False},
    "research":      {"url": "http://localhost:8200/api/research/health", "critical": False},
    "creator":       {"url": "http://localhost:8000/api/creator/health", "critical": False},
    "builder":       {"url": "http://localhost:8443/api/builder/health", "critical": False},
    "finance":       {"url": "http://localhost:8092/api/finance/health", "critical": True},
    "entertainment": {"url": "http://localhost:8400/api/ent/health", "critical": False},
    "writer":        {"url": "http://localhost:8200/api/writer/health", "critical": False},
}

async def check_guild(name: str, cfg: dict) -> dict:
    try:
        r = await httpx.AsyncClient().get(cfg["url"], timeout=10)
        data = r.json()
        status = data.get("status", "unknown")
        statsd.gauge(f"citadel.guild.{name}.healthy", 1 if status == "healthy" else 0)
        return {"guild": name, "status": status, "critical": cfg["critical"]}
    except Exception as e:
        statsd.gauge(f"citadel.guild.{name}.healthy", 0)
        return {"guild": name, "status": "unhealthy", "error": str(e), "critical": cfg["critical"]}

async def run_phase3_health():
    results = await asyncio.gather(*[check_guild(n, c) for n, c in GUILD_SERVICES.items()])
    critical_ok = all(r["status"] == "healthy" for r in results if r["critical"])
    posthog.capture("system", "guild_health_check", {
        "phase": "3.0",
        "critical_healthy": critical_ok,
        "guilds_healthy": sum(1 for r in results if r["status"] == "healthy"),
        "guilds_total": len(results),
    })
    return {"phase": "3.0", "overall": "healthy" if critical_ok else "degraded", "guilds": results}
```

### Datadog Custom Metrics (Phase 3.0)

```python
# tools/dd_metrics_guilds.py
from datadog import statsd

def emit_guild_event(guild: str, event_type: str, properties: dict = {}):
    statsd.increment(f"citadel.guild.{guild}.event.{event_type}")
    for k, v in properties.items():
        if isinstance(v, (int, float)):
            statsd.gauge(f"citadel.guild.{guild}.{k}", v)

def emit_crossguild_flow(source: str, target: str, latency_ms: float, success: bool):
    statsd.increment(f"citadel.crossguild.{source}_to_{target}.total")
    statsd.histogram(f"citadel.crossguild.{source}_to_{target}.latency_ms", latency_ms)
    if not success:
        statsd.increment(f"citadel.crossguild.{source}_to_{target}.failed")

def emit_nats_migration_progress(old_subject: str, new_subject: str, migrated_pct: float):
    statsd.gauge(f"citadel.nats.migration.{new_subject}.pct", migrated_pct)
```
