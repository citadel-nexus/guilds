### NATS Publisher — `tools/bridge/bridge_nats_publisher.py`

```python
#!/usr/bin/env python3
"""NATS Event Publisher for VCC Bridge.

Publishes CRP cycle completion and diagnostic completion events
to the VCC_BRIDGE JetStream stream.

Used by: vcc_cycle_reporter.py (after CRP post)
         perplexity_control_loop_v2.py (after diagnostic WRITE)
"""

import json
import os
import asyncio
import hashlib
from datetime import datetime, timezone
from typing import Optional
from dataclasses import dataclass, asdict

import nats
from nats.js.api import StreamConfig, ConsumerConfig
from pydantic import BaseModel, Field


# ── Environment ──────────────────────────────────────────────────
NATS_URL = os.environ.get("NATS_URL", "nats://localhost:4222")
NATS_CREDS = os.environ.get("NATS_CREDS_FILE", "")
STREAM_NAME = "VCC_BRIDGE"


# ── Payload Schemas ──────────────────────────────────────────────
class CycleCompletedPayload(BaseModel):
    """Published on citadel.vcc.cycle.completed after every CRP post."""
    cycle_id: str                              # e.g. "VCC-FIN-20260301-1200"
    guild: str = "FIN"
    phase: str                                 # e.g. "P1"
    crp_version: str = "2.1"
    guardrail_pass: bool
    drift_clean: bool
    test_passed: int
    test_failed: int
    srs_codes_touched: dict = Field(default_factory=dict)
    health_status: str = "unknown"             # "healthy" | "degraded" | "critical" | "unknown"
    sla_compliant: bool = True
    notion_page_id: Optional[str] = None       # Notion page where CRP was posted
    timestamp: str = Field(
        default_factory=lambda: datetime.now(timezone.utc).isoformat()
    )

    def nats_msg_id(self) -> str:
        """Deterministic message ID for JetStream dedup window."""
        raw = f"{self.cycle_id}:{self.timestamp}"
        return hashlib.sha256(raw.encode()).hexdigest()[:32]


class DiagnosticCompletedPayload(BaseModel):
    """Published on citadel.diagnostic.completed after every Perplexity WRITE."""
    diag_id: str                               # e.g. "DIAG-20260301-1300"
    health_grade: str                          # "HEALTHY" | "DEGRADING" | "CRITICAL" | "UNKNOWN"
    health_score: int = Field(ge=0, le=100)    # 0-100
    blockers: list[str] = Field(default_factory=list)
    recommendations: list[str] = Field(default_factory=list)
    sources_read: int = 0                      # Number of telemetry sources consumed
    l3_verdict: Optional[str] = None           # Bedrock L3 analysis verdict
    infrastructure_status: str = "unknown"     # "stable" | "degraded" | "critical" | "unknown"
    pipeline_pass_rate: Optional[float] = None # 0-100
    nats_connections: Optional[int] = None
    mrr_cents: Optional[int] = None
    active_subscriptions: Optional[int] = None
    triggered_by: str = "cron"                 # "cron" | "nats" | "manual"
    crp_cycle_id: Optional[str] = None         # If triggered by NATS, the source cycle ID
    timestamp: str = Field(
        default_factory=lambda: datetime.now(timezone.utc).isoformat()
    )

    def nats_msg_id(self) -> str:
        raw = f"{self.diag_id}:{self.timestamp}"
        return hashlib.sha256(raw.encode()).hexdigest()[:32]


class BuildPausePayload(BaseModel):
    """Published on citadel.vcc.build.pause when orchestrator pauses VCC."""
    reason: str
    triggered_by: str                          # "orchestrator" | "operator" | "diagnostic"
    diag_id: Optional[str] = None
    health_grade: Optional[str] = None
    timestamp: str = Field(
        default_factory=lambda: datetime.now(timezone.utc).isoformat()
    )


class BuildResumePayload(BaseModel):
    """Published on citadel.vcc.build.resume when orchestrator resumes VCC."""
    reason: str
    triggered_by: str
    diag_id: Optional[str] = None
    health_grade: Optional[str] = None
    timestamp: str = Field(
        default_factory=lambda: datetime.now(timezone.utc).isoformat()
    )


# ── Connection Manager ───────────────────────────────────────────
class BridgeNatsClient:
    """Manages NATS connection and JetStream publishing for the VCC Bridge."""

    def __init__(self, nats_url: str = NATS_URL, creds_file: str = NATS_CREDS):
        self.nats_url = nats_url
        self.creds_file = creds_file
        self._nc: Optional[nats.NATS] = None
        self._js = None

    async def connect(self):
        """Establish NATS connection and bind to JetStream."""
        connect_opts = {"servers": [self.nats_url]}
        if self.creds_file:
            connect_opts["user_credentials"] = self.creds_file

        self._nc = await nats.connect(**connect_opts)
        self._js = self._nc.jetstream()

        # Ensure stream exists
        try:
            await self._js.find_stream_name_by_subject("citadel.vcc.cycle.completed")
        except Exception:
            await self._js.add_stream(
                StreamConfig(
                    name=STREAM_NAME,
                    subjects=[
                        "citadel.vcc.cycle.completed",
                        "citadel.diagnostic.completed",
                        "citadel.vcc.build.pause",
                        "citadel.vcc.build.resume",
                    ],
                    retention="limits",
                    max_msgs=5000,
                    max_age=30 * 24 * 60 * 60,  # 30 days in seconds
                    storage="file",
                )
            )

    async def disconnect(self):
        """Close NATS connection."""
        if self._nc:
            await self._nc.close()

    async def publish_cycle_completed(self, payload: CycleCompletedPayload) -> str:
        """Publish CRP cycle completion event. Returns the stream sequence ID."""
        if not self._js:
            raise RuntimeError("Not connected — call connect() first")

        data = json.dumps(payload.model_dump(), default=str).encode()
        ack = await self._js.publish(
            subject="citadel.vcc.cycle.completed",
            payload=data,
            headers={"Nats-Msg-Id": payload.nats_msg_id()},
        )
        return f"{ack.stream}:{ack.seq}"

    async def publish_diagnostic_completed(self, payload: DiagnosticCompletedPayload) -> str:
        """Publish Perplexity diagnostic completion event."""
        if not self._js:
            raise RuntimeError("Not connected — call connect() first")

        data = json.dumps(payload.model_dump(), default=str).encode()
        ack = await self._js.publish(
            subject="citadel.diagnostic.completed",
            payload=data,
            headers={"Nats-Msg-Id": payload.nats_msg_id()},
        )
        return f"{ack.stream}:{ack.seq}"

    async def publish_build_pause(self, payload: BuildPausePayload) -> str:
        """Publish build pause command."""
        if not self._js:
            raise RuntimeError("Not connected — call connect() first")

        data = json.dumps(payload.model_dump(), default=str).encode()
        ack = await self._js.publish(
            subject="citadel.vcc.build.pause",
            payload=data,
        )
        return f"{ack.stream}:{ack.seq}"

    async def publish_build_resume(self, payload: BuildResumePayload) -> str:
        """Publish build resume command."""
        if not self._js:
            raise RuntimeError("Not connected — call connect() first")

        data = json.dumps(payload.model_dump(), default=str).encode()
        ack = await self._js.publish(
            subject="citadel.vcc.build.resume",
            payload=data,
        )
        return f"{ack.stream}:{ack.seq}"


# ── Convenience function for vcc_cycle_reporter.py integration ───
async def publish_crp_event(
    cycle_id: str,
    phase: str,
    guardrail_pass: bool,
    drift_clean: bool,
    test_passed: int,
    test_failed: int,
    srs_codes_touched: dict,
    health_status: str = "unknown",
    notion_page_id: str | None = None,
) -> str:
    """One-shot publish for CRP integration. Connects, publishes, disconnects."""
    client = BridgeNatsClient()
    try:
        await client.connect()
        payload = CycleCompletedPayload(
            cycle_id=cycle_id,
            phase=phase,
            guardrail_pass=guardrail_pass,
            drift_clean=drift_clean,
            test_passed=test_passed,
            test_failed=test_failed,
            srs_codes_touched=srs_codes_touched,
            health_status=health_status,
            notion_page_id=notion_page_id,
        )
        seq = await client.publish_cycle_completed(payload)
        return seq
    finally:
        await client.disconnect()


def publish_crp_event_sync(**kwargs) -> str:
    """Synchronous wrapper for publish_crp_event()."""
    return asyncio.get_event_loop().run_until_complete(publish_crp_event(**kwargs))
```

## BRIDGE-002: CRP → Perplexity Trigger


### NATS Subscriber — `tools/bridge/bridge_crp_trigger.py`

```python
#!/usr/bin/env python3
"""CRP → Perplexity Trigger — NATS subscriber that triggers Perplexity loop
on VCC cycle completion.

Listens on: citadel.vcc.cycle.completed
Action: Runs Perplexity Control Loop v2 READ→RAG→THINK→WRITE pipeline
Fallback: Cron fires if no NATS event within 90 minutes

Usage:
    # As persistent listener (production)
    python tools/bridge/bridge_crp_trigger.py --listen

    # As one-shot check (CI/testing)
    python tools/bridge/bridge_crp_trigger.py --check-last
"""

import asyncio
import json
import os
import logging
import time
from datetime import datetime, timezone, timedelta
from typing import Optional

import nats
from nats.js.api import ConsumerConfig, DeliverPolicy, AckPolicy

from tools.bridge.bridge_nats_publisher import CycleCompletedPayload

log = logging.getLogger("bridge.crp_trigger")
log.setLevel(logging.INFO)

NATS_URL = os.environ.get("NATS_URL", "nats://localhost:4222")
NATS_CREDS = os.environ.get("NATS_CREDS_FILE", "")
CRON_FALLBACK_MINUTES = int(os.environ.get("BRIDGE_CRON_FALLBACK_MINUTES", "90"))
DEDUP_WINDOW_SECONDS = int(os.environ.get("BRIDGE_DEDUP_WINDOW_SECONDS", "60"))
STREAM_NAME = "VCC_BRIDGE"
CONSUMER_NAME = "perplexity-crp-trigger"


class CrpTriggerListener:
    """Listens for CRP cycle completions and triggers Perplexity loop."""

    def __init__(self):
        self._nc: Optional[nats.NATS] = None
        self._js = None
        self._sub = None
        self._last_trigger_time: float = 0
        self._last_cycle_id: str = ""
        self._running = True

    async def connect(self):
        connect_opts = {"servers": [NATS_URL]}
        if NATS_CREDS:
            connect_opts["user_credentials"] = NATS_CREDS
        self._nc = await nats.connect(**connect_opts)
        self._js = self._nc.jetstream()
        log.info(f"Connected to NATS at {NATS_URL}")

    async def disconnect(self):
        if self._sub:
            await self._sub.unsubscribe()
        if self._nc:
            await self._nc.close()

    def _should_dedup(self, cycle_id: str) -> bool:
        """Reject duplicate triggers within the dedup window."""
        now = time.time()
        if (
            cycle_id == self._last_cycle_id
            and (now - self._last_trigger_time) < DEDUP_WINDOW_SECONDS
        ):
            log.info(f"[DEDUP] Ignoring duplicate cycle {cycle_id} within {DEDUP_WINDOW_SECONDS}s window")
            return True
        return False

    async def _trigger_perplexity(self, payload: CycleCompletedPayload, trigger_source: str):
        """Run the Perplexity Control Loop v2 with CRP context injection."""
        log.info(f"[TRIGGER] {trigger_source}: {payload.cycle_id}")

        # Update dedup tracking
        self._last_trigger_time = time.time()
        self._last_cycle_id = payload.cycle_id

        # Build CRP context for Perplexity system prompt
        crp_context = {
            "source": "vcc_crp",
            "cycle_id": payload.cycle_id,
            "guild": payload.guild,
            "phase": payload.phase,
            "guardrail_pass": payload.guardrail_pass,
            "drift_clean": payload.drift_clean,
            "test_passed": payload.test_passed,
            "test_failed": payload.test_failed,
            "srs_codes_touched": payload.srs_codes_touched,
            "health_status": payload.health_status,
            "sla_compliant": payload.sla_compliant,
        }

        crp_prompt_injection = (
            f"The VCC just completed cycle {payload.cycle_id} in phase {payload.phase}. "
            f"Guardrails: {'PASS' if payload.guardrail_pass else 'FAIL'}. "
            f"Tests: {payload.test_passed} passed, {payload.test_failed} failed. "
            f"Drift: {'clean' if payload.drift_clean else 'DIRTY'}. "
            f"SLA: {'compliant' if payload.sla_compliant else 'BREACH'}. "
            f"SRS codes touched: {list(payload.srs_codes_touched.keys())}. "
            f"Factor this deployment into your infrastructure analysis — "
            f"check if new code introduced regressions."
        )

        # Import and run Perplexity loop with injected context
        try:
            from tools.perplexity_control_loop_v2 import run_loop
            await run_loop(
                extra_sources=[crp_context],
                system_prompt_append=crp_prompt_injection,
                triggered_by=trigger_source,
                crp_cycle_id=payload.cycle_id,
            )
            log.info(f"[COMPLETE] Perplexity loop finished for {payload.cycle_id}")
        except ImportError:
            log.warning("perplexity_control_loop_v2 not available — running stub")
            # Stub for development/testing
            log.info(f"[STUB] Would run Perplexity loop with context: {crp_context}")
        except Exception as e:
            log.error(f"[ERROR] Perplexity loop failed: {e}")
            raise

    async def _handle_message(self, msg):
        """Process a single NATS message."""
        try:
            data = json.loads(msg.data.decode())
            payload = CycleCompletedPayload(**data)

            if self._should_dedup(payload.cycle_id):
                await msg.ack()
                return

            await self._trigger_perplexity(payload, trigger_source="NATS: citadel.vcc.cycle.completed")
            await msg.ack()
        except Exception as e:
            log.error(f"[ERROR] Failed to process message: {e}")
            await msg.nak(delay=10)  # Retry after 10s

    async def listen(self):
        """Start persistent listener with cron fallback."""
        await self.connect()

        # Subscribe to JetStream
        self._sub = await self._js.pull_subscribe(
            subject="citadel.vcc.cycle.completed",
            durable=CONSUMER_NAME,
            stream=STREAM_NAME,
        )
        log.info(f"Subscribed to citadel.vcc.cycle.completed (consumer: {CONSUMER_NAME})")

        while self._running:
            try:
                # Fetch with timeout — this is the event-driven path
                msgs = await self._sub.fetch(batch=1, timeout=CRON_FALLBACK_MINUTES * 60)
                for msg in msgs:
                    await self._handle_message(msg)
            except nats.errors.TimeoutError:
                # No message received within cron fallback window — fire cron
                log.info(f"[TRIGGER] CRON: {CRON_FALLBACK_MINUTES}min timeout — no NATS event")
                fallback_payload = CycleCompletedPayload(
                    cycle_id="CRON-FALLBACK",
                    phase="unknown",
                    guardrail_pass=True,
                    drift_clean=True,
                    test_passed=0,
                    test_failed=0,
                    health_status="unknown",
                )
                await self._trigger_perplexity(fallback_payload, trigger_source="CRON: 90min timeout")
            except Exception as e:
                log.error(f"[ERROR] Listener error: {e}")
                await asyncio.sleep(5)  # Brief pause before retry

    async def check_last(self) -> dict:
        """Check the last received message (for CI/testing)."""
        await self.connect()
        try:
            sub = await self._js.pull_subscribe(
                subject="citadel.vcc.cycle.completed",
                durable=f"{CONSUMER_NAME}-check",
                stream=STREAM_NAME,
                config=ConsumerConfig(
                    deliver_policy=DeliverPolicy.LAST,
                    ack_policy=AckPolicy.NONE,
                ),
            )
            msgs = await sub.fetch(batch=1, timeout=5)
            if msgs:
                data = json.loads(msgs[0].data.decode())
                return {"found": True, "payload": data}
            return {"found": False}
        finally:
            await self.disconnect()

    def stop(self):
        self._running = False


if __name__ == "__main__":
    import argparse
    logging.basicConfig(level=logging.INFO, format="%(asctime)s %(name)s %(message)s")
    parser = argparse.ArgumentParser(description="CRP → Perplexity Trigger")
    parser.add_argument("--listen", action="store_true", help="Start persistent listener")
    parser.add_argument("--check-last", action="store_true", help="Check last message")
    args = parser.parse_args()

    listener = CrpTriggerListener()
    if args.listen:
        asyncio.run(listener.listen())
```

### Implementation — `tools/bridge/bridge_feedback_injection.py`

```python
#!/usr/bin/env python3
"""Perplexity → CRP Feedback Injection.

Reads the latest Perplexity diagnostic from public.vcc_loop_state
and injects it into CRP §7 as subsections §7.4, §7.5, and §7.6.

Called by: vcc_cycle_reporter.py during §7 assembly
"""

import json
import os
import logging
from dataclasses import dataclass, field, asdict
from datetime import datetime, timezone, timedelta
from typing import Optional

import requests

log = logging.getLogger("bridge.feedback_injection")

# ── Config ───────────────────────────────────────────────────────
SUPABASE_URL = os.environ.get("SUPABASE_URL", "")
SUPABASE_SERVICE_KEY = os.environ.get("SUPABASE_SERVICE_ROLE_KEY", "")
NOTION_API = "https://api.notion.com/v1"
NOTION_TOKEN = os.environ.get("NOTION_INTERNAL_TOKEN", "")
NOTION_VERSION = "2022-06-28"
PERPLEXITY_DIAGNOSTIC_DB_ID = os.environ.get("PERPLEXITY_DIAGNOSTIC_DB_ID", "")

STALE_THRESHOLD_HOURS = int(os.environ.get("BRIDGE_STALE_DIAGNOSTIC_HOURS", "6"))
GRACE_PERIOD_CYCLES = int(os.environ.get("BRIDGE_GRACE_PERIOD_CYCLES", "3"))


@dataclass
class DiagnosticSummary:
    """Extracted summary from last Perplexity diagnostic."""
    available: bool = False
    diag_id: str = ""
    health_grade: str = "UNKNOWN"
    health_score: int = 0
    blockers: list = field(default_factory=list)
    recommendations: list = field(default_factory=list)
    infrastructure_status: str = "unknown"
    pipeline_pass_rate: Optional[float] = None
    nats_connections: Optional[int] = None
    mrr_cents: Optional[int] = None
    active_subscriptions: Optional[int] = None
    l3_verdict: str = ""
    timestamp: str = ""
    stale: bool = False               # True if diagnostic is older than STALE_THRESHOLD_HOURS
    source: str = "none"              # "state_store" | "notion_api" | "none"


def read_last_diagnostic() -> DiagnosticSummary:
    """Read the latest Perplexity diagnostic from vcc_loop_state.

    Fallback chain:
    1. Query public.vcc_loop_state for latest diagnostic row
    2. If state store unavailable, query Notion diagnostic page
    3. If both fail, return empty summary with available=False
    """
    # Attempt 1: State store (preferred — fastest, most structured)
    summary = _read_from_state_store()
    if summary.available:
        return summary

    # Attempt 2: Notion API fallback
    log.info("State store unavailable — falling back to Notion API")
    summary = _read_from_notion()
    if summary.available:
        return summary

    # Attempt 3: No diagnostic available
    log.warning("No Perplexity diagnostic available from any source")
    return DiagnosticSummary(
        available=False,
        health_grade="UNKNOWN",
        source="none",
    )


def _read_from_state_store() -> DiagnosticSummary:
    """Query public.vcc_loop_state for latest diagnostic."""
    if not SUPABASE_URL or not SUPABASE_SERVICE_KEY:
        return DiagnosticSummary(available=False)

    try:
        headers = {
            "apikey": SUPABASE_SERVICE_KEY,
            "Authorization": f"Bearer {SUPABASE_SERVICE_KEY}",
            "Content-Type": "application/json",
        }
        resp = requests.get(
            f"{SUPABASE_URL}/rest/v1/vcc_loop_state",
            headers=headers,
            params={
                "loop_name": "eq.perplexity_v2",
                "order": "updated_at.desc",
                "limit": "1",
            },
            timeout=10,
        )
        resp.raise_for_status()
        rows = resp.json()
        if not rows:
            return DiagnosticSummary(available=False)

        row = rows[0]
        state = row.get("state_json", {})
        if isinstance(state, str):
            state = json.loads(state)

        # Check staleness
        updated_at = row.get("updated_at", "")
        stale = False
        if updated_at:
            try:
                updated = datetime.fromisoformat(updated_at.replace("Z", "+00:00"))
                stale = (datetime.now(timezone.utc) - updated) > timedelta(hours=STALE_THRESHOLD_HOURS)
            except (ValueError, TypeError):
                pass

        return DiagnosticSummary(
            available=True,
            diag_id=state.get("diag_id", row.get("last_run_id", "")),
            health_grade=state.get("health_grade", "UNKNOWN"),
            health_score=int(state.get("health_score", 0)),
            blockers=state.get("blockers", []),
            recommendations=state.get("recommendations", [])[:5],
            infrastructure_status=state.get("infrastructure_status", "unknown"),
            pipeline_pass_rate=state.get("pipeline_pass_rate"),
            nats_connections=state.get("nats_connections"),
            mrr_cents=state.get("mrr_cents"),
            active_subscriptions=state.get("active_subscriptions"),
            l3_verdict=state.get("l3_verdict", ""),
            timestamp=updated_at,
            stale=stale,
            source="state_store",
        )
    except Exception as e:
        log.error(f"State store read failed: {e}")
        return DiagnosticSummary(available=False)


def _read_from_notion() -> DiagnosticSummary:
    """Fallback: read latest diagnostic from Notion page."""
    if not NOTION_TOKEN or not PERPLEXITY_DIAGNOSTIC_DB_ID:
        return DiagnosticSummary(available=False)

    try:
        headers = {
            "Authorization": f"Bearer {NOTION_TOKEN}",
            "Notion-Version": NOTION_VERSION,
        }
        resp = requests.post(
            f"{NOTION_API}/databases/{PERPLEXITY_DIAGNOSTIC_DB_ID}/query",
            headers=headers,
            json={
                "sorts": [{"timestamp": "created_time", "direction": "descending"}],
                "page_size": 1,
            },
            timeout=15,
        )
        resp.raise_for_status()
        results = resp.json().get("results", [])
        if not results:
            return DiagnosticSummary(available=False)

        page = results[0]
        props = page.get("properties", {})

        # Extract fields from Notion page properties
        def _get_text(prop_name: str) -> str:
            prop = props.get(prop_name, {})
            if prop.get("type") == "rich_text":
                return "".join(t.get("plain_text", "") for t in prop.get("rich_text", []))
            if prop.get("type") == "title":
                return "".join(t.get("plain_text", "") for t in prop.get("title", []))
            return ""

        def _get_number(prop_name: str) -> Optional[int]:
            prop = props.get(prop_name, {})
            if prop.get("type") == "number":
                return prop.get("number")
            return None

        return DiagnosticSummary(
            available=True,
            diag_id=_get_text("Diagnostic ID") or page["id"][:8],
            health_grade=_get_text("Health Grade") or "UNKNOWN",
            health_score=_get_number("Health Score") or 0,
            blockers=[],  # Would need to parse page content for blockers
            timestamp=page.get("created_time", ""),
            source="notion_api",
        )
    except Exception as e:
        log.error(f"Notion API read failed: {e}")
        return DiagnosticSummary(available=False)


# ── CRP Injection Helpers ────────────────────────────────────────

def build_crp_section_7_4(diag: DiagnosticSummary) -> dict:
    """Build CRP §7.4 — Perplexity Diagnostic (last run)."""
    if not diag.available:
        return {
            "section": "§7.4 — Perplexity Diagnostic",
            "status": "unavailable",
            "message": "No Perplexity diagnostic available yet",
            "blocking": False,
        }

    stale_warning = " ⚠️ STALE" if diag.stale else ""
    return {
        "section": "§7.4 — Perplexity Diagnostic (last run)",
        "diag_id": diag.diag_id,
        "health_grade": f"{diag.health_grade}{stale_warning}",
        "health_score": diag.health_score,
        "infrastructure_status": diag.infrastructure_status,
        "top_blockers": diag.blockers[:3],
        "l3_verdict": diag.l3_verdict or "N/A",
        "source": diag.source,
        "timestamp": diag.timestamp,
    }


def build_crp_section_7_5(diag: DiagnosticSummary) -> dict:
    """Build CRP §7.5 — Bridge Status."""
    return {
        "section": "§7.5 — Bridge Status",
        "bridge_active": diag.available,
        "data_source": diag.source,
        "last_diagnostic_age": diag.timestamp,
        "stale": diag.stale,
        "pipeline_pass_rate": diag.pipeline_pass_rate,
        "nats_connections": diag.nats_connections,
    }


def build_crp_section_7_6(diag: DiagnosticSummary) -> dict:
    """Build CRP §7.6 — Merged Health."""
    return {
        "section": "§7.6 — Merged Health",
        "perplexity_score": diag.health_score if diag.available else None,
        "perplexity_grade": diag.health_grade if diag.available else "UNKNOWN",
        "mrr_cents": diag.mrr_cents,
        "active_subscriptions": diag.active_subscriptions,
        "recommendations": diag.recommendations[:3] if diag.available else [],
    }


def evaluate_crp_impact(diag: DiagnosticSummary) -> dict:
    """Determine how the diagnostic affects CRP status.

    Returns:
        dict with 'status' ("GO", "WARNING", "NO-GO"),
        'reason' string, and 'block' boolean.
    """
    if not diag.available:
        return {
            "status": "GO",
            "reason": "No diagnostic available — grace period active",
            "block": False,
        }

    if diag.health_grade == "CRITICAL":
        return {
            "status": "NO-GO",
            "reason": (
                f"Perplexity diagnostic reports CRITICAL infrastructure state "
                f"(score: {diag.health_score}/100, diag: {diag.diag_id}) — "
                f"do not deploy"
            ),
            "block": True,
        }

    if diag.health_grade == "DEGRADING" and diag.health_score < 60:
        return {
            "status": "WARNING",
            "reason": (
                f"Infrastructure degrading (score: {diag.health_score}/100) — "
                f"prioritize Perplexity blockers before continuing build order. "
                f"Blockers: {', '.join(diag.blockers[:3]) if diag.blockers else 'none listed'}"
            ),
            "block": False,
        }

    if diag.stale:
        return {
            "status": "WARNING",
            "reason": f"Perplexity diagnostic is stale (>{STALE_THRESHOLD_HOURS}h old) — results may not reflect current state",
            "block": False,
        }

    return {
        "status": "GO",
        "reason": f"Perplexity diagnostic: {diag.health_grade} ({diag.health_score}/100)",
        "block": False,
    }
```

## BRIDGE-004: Unified Health Schema


### Implementation — `tools/bridge/bridge_health_schema.py`

```python
#!/usr/bin/env python3
"""Unified Health Schema — CitadelHealthSnapshot.

Single Pydantic model that both VCC (CRP) and Perplexity loops populate.
The loop_orchestrator.py merges partial snapshots into a 'source: merged' snapshot.

Used by: vcc_cycle_reporter.py, perplexity_control_loop_v2.py, loop_orchestrator.py
"""

from __future__ import annotations

from datetime import datetime, timezone
from typing import Literal, Optional
from pydantic import BaseModel, Field


# ── Domain Section Models ────────────────────────────────────────

class ServiceHealth(BaseModel):
    """Per-service health from /health endpoints."""
    service_id: str                          # e.g. "smp-fin-api"
    status: Literal["healthy", "degraded", "unhealthy", "unknown"] = "unknown"
    response_time_ms: Optional[float] = None
    consecutive_failures: int = 0
    last_check: Optional[str] = None         # ISO 8601
    dependencies: dict[str, str] = Field(default_factory=dict)
    # e.g. {"supabase": "connected", "nats": "connected", "redis": "connected"}


class InfrastructureHealth(BaseModel):
    """Infrastructure metrics from Datadog / CloudWatch."""
    cpu_pct: Optional[float] = None
    memory_pct: Optional[float] = None
    disk_pct: Optional[float] = None
    nats_connections: Optional[int] = None
    nats_total_connections: Optional[int] = None
    ecs_running_tasks: Optional[int] = None
    ecs_desired_tasks: Optional[int] = None
    active_monitors: int = 0
    alerting_monitors: int = 0
    supabase_pool_used: Optional[int] = None
    supabase_pool_max: Optional[int] = None
    redis_memory_used_mb: Optional[float] = None
    redis_connected_clients: Optional[int] = None


class CodeHealth(BaseModel):
    """Code quality metrics from GitLab + pytest."""
    pipeline_pass_rate: Optional[float] = None  # 0-100
    deploy_frequency_daily: Optional[float] = None
    lead_time_hours: Optional[float] = None
    change_failure_rate: Optional[float] = None  # 0-100
    mttr_hours: Optional[float] = None           # Mean time to recovery
    test_count: int = 0
    test_passed: int = 0
    test_failed: int = 0
    coverage_pct: Optional[float] = None
    guardrail_violations: int = 0
    drift_items: int = 0
    bundle_size_kb: Optional[float] = None
    build_time_seconds: Optional[float] = None


class RevenueHealth(BaseModel):
    """Revenue metrics from Stripe."""
    mrr_cents: int = 0
    arr_cents: int = 0
    active_subscriptions: int = 0
    trialing_subscriptions: int = 0
    past_due_subscriptions: int = 0
    churn_rate_pct: Optional[float] = None
    total_customers: int = 0


# ── Composite Snapshot ───────────────────────────────────────────

class CitadelHealthSnapshot(BaseModel):
    """Unified health snapshot — single source of truth for both loops.

    Source values:
    - "crp": Populated by VCC cycle reporter
    - "perplexity": Populated by Perplexity control loop v2
    - "merged": Merged by loop_orchestrator.py from both sources
    """
    snapshot_id: str                         # "HEALTH-{YYYYMMDD}-{HHMM}"
    timestamp: str = Field(
        default_factory=lambda: datetime.now(timezone.utc).isoformat()
    )
    source: Literal["crp", "perplexity", "merged"]

    # ── Composite Grade ──────────────────────────────────────────
    overall_grade: Literal["HEALTHY", "DEGRADING", "CRITICAL", "UNKNOWN"] = "UNKNOWN"
    overall_score: int = Field(default=0, ge=0, le=100)

    # ── Domain Sections ──────────────────────────────────────────
    services: list[ServiceHealth] = Field(default_factory=list)
    infrastructure: Optional[InfrastructureHealth] = None
    code: Optional[CodeHealth] = None
    revenue: Optional[RevenueHealth] = None

    # ── CRP-specific Fields ──────────────────────────────────────
    crp_cycle_id: Optional[str] = None
    crp_version: Optional[str] = None
    guardrail_pass: Optional[bool] = None
    drift_clean: Optional[bool] = None
    sla_compliant: Optional[bool] = None
    srs_codes_touched: dict = Field(default_factory=dict)

    # ── Perplexity-specific Fields ───────────────────────────────
    diag_id: Optional[str] = None
    perplexity_score: Optional[int] = None
    l3_verdict: Optional[str] = None
    blockers: list[str] = Field(default_factory=list)
    recommendations: list[str] = Field(default_factory=list)
    triggered_by: Optional[str] = None


def compute_overall_grade(
    services: list[ServiceHealth],
    infrastructure: Optional[InfrastructureHealth],
    code: Optional[CodeHealth],
    perplexity_score: Optional[int],
) -> tuple[str, int]:
    """Compute overall grade and score from domain sections.

    Scoring:
    - Services: 30% weight (based on % healthy)
    - Infrastructure: 25% weight
    - Code: 25% weight
    - Perplexity: 20% weight (if available)

    Grade thresholds:
    - HEALTHY: >= 75
    - DEGRADING: >= 50
    - CRITICAL: < 50
    - UNKNOWN: no data
    """
    scores = []
    weights = []

    # Services score (30%)
    if services:
        healthy = sum(1 for s in services if s.status == "healthy")
        svc_score = (healthy / len(services)) * 100 if services else 0
        scores.append(svc_score)
        weights.append(0.30)

    # Infrastructure score (25%)
    if infrastructure:
        infra_score = 100.0
        if infrastructure.cpu_pct and infrastructure.cpu_pct > 80:
            infra_score -= 20
        if infrastructure.memory_pct and infrastructure.memory_pct > 85:
            infra_score -= 20
        if infrastructure.alerting_monitors > 0:
            infra_score -= min(30, infrastructure.alerting_monitors * 10)
        scores.append(max(0, infra_score))
        weights.append(0.25)

    # Code score (25%)
    if code:
        code_score = 100.0
        if code.test_failed > 0:
            code_score -= min(30, code.test_failed * 5)
        if code.guardrail_violations > 0:
            code_score -= min(30, code.guardrail_violations * 10)
        if code.drift_items > 0:
            code_score -= min(20, code.drift_items * 5)
        if code.coverage_pct and code.coverage_pct < 80:
            code_score -= 10
        scores.append(max(0, code_score))
        weights.append(0.25)

    # Perplexity score (20%)
    if perplexity_score is not None:
        scores.append(float(perplexity_score))
        weights.append(0.20)

    if not scores:
        return ("UNKNOWN", 0)

    # Normalize weights
    total_weight = sum(weights)
    overall = sum(s * w for s, w in zip(scores, weights)) / total_weight
    overall = int(round(overall))

    # Grade
    if overall >= 75:
        grade = "HEALTHY"
    elif overall >= 50:
        grade = "DEGRADING"
    else:
        grade = "CRITICAL"

    return (grade, overall)


def merge_snapshots(
    crp_snapshot: Optional[CitadelHealthSnapshot],
    perplexity_snapshot: Optional[CitadelHealthSnapshot],
) -> CitadelHealthSnapshot:
    """Merge CRP and Perplexity snapshots into a unified merged snapshot.

    Rules:
    - CRP populates: services, code.test_*, guardrail_pass, drift_clean, sla_compliant
    - Perplexity populates: infrastructure, code.pipeline_*, revenue, blockers, recommendations
    - Merged: worst-of-both for grade, combined for all sections
    """
    now = datetime.now(timezone.utc)
    snapshot_id = f"HEALTH-{now.strftime('%Y%m%d-%H%M')}"

    # Start with empty merged snapshot
    merged = CitadelHealthSnapshot(
        snapshot_id=snapshot_id,
        source="merged",
    )

    # Merge CRP fields
    if crp_snapshot:
        merged.services = crp_snapshot.services
        merged.crp_cycle_id = crp_snapshot.crp_cycle_id
        merged.crp_version = crp_snapshot.crp_version
        merged.guardrail_pass = crp_snapshot.guardrail_pass
        merged.drift_clean = crp_snapshot.drift_clean
        merged.sla_compliant = crp_snapshot.sla_compliant
        merged.srs_codes_touched = crp_snapshot.srs_codes_touched
        if crp_snapshot.code:
            merged.code = crp_snapshot.code

    # Merge Perplexity fields
    if perplexity_snapshot:
        merged.infrastructure = perplexity_snapshot.infrastructure
        merged.revenue = perplexity_snapshot.revenue
        merged.diag_id = perplexity_snapshot.diag_id
        merged.perplexity_score = perplexity_snapshot.perplexity_score
        merged.l3_verdict = perplexity_snapshot.l3_verdict
        merged.blockers = perplexity_snapshot.blockers
        merged.recommendations = perplexity_snapshot.recommendations
        merged.triggered_by = perplexity_snapshot.triggered_by
        # Merge Perplexity code metrics into existing CRP code metrics
        if perplexity_snapshot.code:
            if merged.code is None:
                merged.code = perplexity_snapshot.code
            else:
                # Perplexity owns DORA metrics, CRP owns test metrics
                merged.code.pipeline_pass_rate = perplexity_snapshot.code.pipeline_pass_rate
                merged.code.deploy_frequency_daily = perplexity_snapshot.code.deploy_frequency_daily
                merged.code.lead_time_hours = perplexity_snapshot.code.lead_time_hours
                merged.code.change_failure_rate = perplexity_snapshot.code.change_failure_rate
                merged.code.mttr_hours = perplexity_snapshot.code.mttr_hours

    # Compute overall grade from merged data
    grade, score = compute_overall_grade(
        merged.services,
        merged.infrastructure,
        merged.code,
        merged.perplexity_score,
    )
    merged.overall_grade = grade
    merged.overall_score = score

    return merged
```

## BRIDGE-005: Shared Dedup & Conflict Resolution


### Implementation — `tools/bridge/issue_dedup.py`

```python
#!/usr/bin/env python3
"""Shared Issue Dedup — Prevents VCC and Perplexity from filing duplicate issues.

Checks Linear + GitLab before creating new issues.
Used by: vcc_cycle_reporter.py (when filing drift/guardrail issues)
         perplexity_control_loop_v2.py (when filing infrastructure issues)
"""

import os
import re
import json
import hashlib
import logging
from dataclasses import dataclass, field, asdict
from datetime import datetime, timezone, timedelta
from typing import Optional

import requests

log = logging.getLogger("bridge.issue_dedup")

# ── Config ───────────────────────────────────────────────────────
LINEAR_API_URL = "https://api.linear.app/graphql"
LINEAR_API_KEY = os.environ.get("LINEAR_API_KEY", "")
GITLAB_API_URL = os.environ.get("GITLAB_API_URL", "https://gitlab.com/api/v4")
GITLAB_TOKEN = os.environ.get("GITLAB_TOKEN", "")
GITLAB_PROJECT_ID = os.environ.get("GITLAB_PROJECT_ID", "")
DEDUP_LOOKBACK_HOURS = int(os.environ.get("BRIDGE_DEDUP_LOOKBACK_HOURS", "72"))


@dataclass
class ExistingIssue:
    """An existing issue found by dedup search."""
    source: str           # "linear" | "gitlab"
    issue_id: str
    title: str
    state: str            # "open" | "closed" | "in_progress"
    created_at: str
    labels: list = field(default_factory=list)
    url: str = ""
    similarity_score: float = 0.0  # 0-1, how similar to the new issue


@dataclass
class DedupResult:
    """Result of dedup check."""
    is_duplicate: bool
    existing_issues: list = field(default_factory=list)  # list[ExistingIssue]
    fingerprint: str = ""          # Hash of the issue content for future dedup
    recommended_action: str = ""   # "skip" | "comment" | "create"


def compute_issue_fingerprint(title: str, description: str = "", labels: list = None) -> str:
    """Compute a deterministic fingerprint for an issue.

    Strips common prefixes/suffixes, normalizes whitespace, and hashes.
    Same underlying problem should produce the same fingerprint regardless
    of which loop created it.
    """
    # Normalize
    text = f"{title} {description}".lower().strip()
    # Remove common prefixes from both loops
    prefixes = [
        r"^\[vcc\]\s*", r"^\[perplexity\]\s*", r"^\[crp\]\s*",
        r"^\[diag\]\s*", r"^\[bridge\]\s*",
        r"^fix:\s*", r"^bug:\s*", r"^alert:\s*",
    ]
    for prefix in prefixes:
        text = re.sub(prefix, "", text, flags=re.IGNORECASE)
    # Remove timestamps and IDs
    text = re.sub(r"\d{4}-\d{2}-\d{2}[T\s]\d{2}:\d{2}[:\d.Z]*", "", text)
    text = re.sub(r"VCC-FIN-\d{8}-\d{4}", "", text)
    text = re.sub(r"DIAG-\d{8}-\d{4}", "", text)
    # Normalize whitespace
    text = re.sub(r"\s+", " ", text).strip()
    return hashlib.sha256(text.encode()).hexdigest()[:16]


def check_linear_duplicates(title: str, labels: list = None) -> list[ExistingIssue]:
    """Search Linear for potential duplicate issues."""
    if not LINEAR_API_KEY:
        return []

    lookback = datetime.now(timezone.utc) - timedelta(hours=DEDUP_LOOKBACK_HOURS)
    # Extract key terms for search
    search_terms = re.sub(r"[\[\](){}]", "", title)[:100]

    query = """
    query SearchIssues($filter: IssueFilter, $term: String) {
        issueSearch(filter: $filter, term: $term, first: 10) {
            nodes {
                id
                identifier
                title
                state { name }
                createdAt
                labels { nodes { name } }
                url
            }
        }
    }
    """
    variables = {
        "term": search_terms,
        "filter": {
            "createdAt": {"gte": lookback.isoformat()},
        },
    }

    try:
        resp = requests.post(
            LINEAR_API_URL,
            headers={"Authorization": LINEAR_API_KEY, "Content-Type": "application/json"},
            json={"query": query, "variables": variables},
            timeout=10,
        )
        resp.raise_for_status()
        data = resp.json()
        nodes = data.get("data", {}).get("issueSearch", {}).get("nodes", [])

        results = []
        new_fp = compute_issue_fingerprint(title)
        for node in nodes:
            existing_fp = compute_issue_fingerprint(node["title"])
            # Fingerprint match = high confidence duplicate
            similarity = 1.0 if existing_fp == new_fp else _title_similarity(title, node["title"])
            if similarity >= 0.6:
                results.append(ExistingIssue(
                    source="linear",
                    issue_id=node["identifier"],
                    title=node["title"],
                    state=node["state"]["name"].lower() if node.get("state") else "unknown",
                    created_at=node["createdAt"],
                    labels=[l["name"] for l in node.get("labels", {}).get("nodes", [])],
                    url=node.get("url", ""),
                    similarity_score=similarity,
                ))
        return results
    except Exception as e:
        log.error(f"Linear search failed: {e}")
        return []


def check_gitlab_duplicates(title: str, labels: list = None) -> list[ExistingIssue]:
    """Search GitLab for potential duplicate issues."""
    if not GITLAB_TOKEN or not GITLAB_PROJECT_ID:
        return []

    lookback = datetime.now(timezone.utc) - timedelta(hours=DEDUP_LOOKBACK_HOURS)
    search_terms = re.sub(r"[\[\](){}]", "", title)[:100]

    try:
        resp = requests.get(
            f"{GITLAB_API_URL}/projects/{GITLAB_PROJECT_ID}/issues",
            headers={"PRIVATE-TOKEN": GITLAB_TOKEN},
            params={
                "search": search_terms,
                "created_after": lookback.isoformat(),
                "per_page": 10,
                "state": "opened",
            },
            timeout=10,
        )
        resp.raise_for_status()
        issues = resp.json()

        results = []
        new_fp = compute_issue_fingerprint(title)
        for issue in issues:
            existing_fp = compute_issue_fingerprint(issue["title"])
            similarity = 1.0 if existing_fp == new_fp else _title_similarity(title, issue["title"])
            if similarity >= 0.6:
                results.append(ExistingIssue(
                    source="gitlab",
                    issue_id=str(issue["iid"]),
                    title=issue["title"],
                    state="open" if issue["state"] == "opened" else issue["state"],
                    created_at=issue["created_at"],
                    labels=issue.get("labels", []),
                    url=issue.get("web_url", ""),
                    similarity_score=similarity,
                ))
        return results
    except Exception as e:
        log.error(f"GitLab search failed: {e}")
        return []


def _title_similarity(a: str, b: str) -> float:
    """Simple Jaccard similarity on word tokens."""
    words_a = set(a.lower().split())
    words_b = set(b.lower().split())
    if not words_a or not words_b:
        return 0.0
    intersection = words_a & words_b
    union = words_a | words_b
    return len(intersection) / len(union)


def check_duplicate(
    title: str,
    description: str = "",
    labels: list = None,
    sources: list = None,
) -> DedupResult:
    """Full dedup check across all configured sources.

    Args:
        title: Issue title
        description: Issue body (optional, used for fingerprinting)
        labels: Issue labels
        sources: Which sources to check (["linear", "gitlab"]). Default: both.

    Returns:
        DedupResult with is_duplicate flag and recommended action.
    """
    if sources is None:
        sources = ["linear", "gitlab"]

    fingerprint = compute_issue_fingerprint(title, description, labels)
    existing = []

    if "linear" in sources:
        existing.extend(check_linear_duplicates(title, labels))
    if "gitlab" in sources:
        existing.extend(check_gitlab_duplicates(title, labels))

    # Sort by similarity score descending
    existing.sort(key=lambda x: x.similarity_score, reverse=True)

    if not existing:
        return DedupResult(
            is_duplicate=False,
            fingerprint=fingerprint,
            recommended_action="create",
        )

    # If top match has similarity >= 0.8, it's a duplicate
    top_match = existing[0]
    if top_match.similarity_score >= 0.8:
        if top_match.state in ("open", "in_progress"):
            return DedupResult(
                is_duplicate=True,
                existing_issues=[asdict(e) for e in existing[:3]],
                fingerprint=fingerprint,
                recommended_action="comment",  # Add a comment to existing issue
            )
        else:
            # Issue was closed — might need reopening
            return DedupResult(
                is_duplicate=True,
                existing_issues=[asdict(e) for e in existing[:3]],
                fingerprint=fingerprint,
                recommended_action="create",  # Create new — old one is closed
            )

    # Moderate similarity — flag but allow creation
    return DedupResult(
        is_duplicate=False,
        existing_issues=[asdict(e) for e in existing[:3]],
        fingerprint=fingerprint,
        recommended_action="create",
    )
```

```python
#!/usr/bin/env python3
"""Shared State Store Client — Read/write to public.vcc_loop_state.

Used by: vcc_cycle_reporter.py, perplexity_control_loop_v2.py, loop_orchestrator.py
"""

import json
import os
import logging
from datetime import datetime, timezone
from typing import Optional
from dataclasses import dataclass, asdict

import requests

log = logging.getLogger("bridge.state_store")

SUPABASE_URL = os.environ.get("SUPABASE_URL", "")
SUPABASE_SERVICE_KEY = os.environ.get("SUPABASE_SERVICE_ROLE_KEY", "")


@dataclass
class LoopState:
    loop_name: str
    status: str = "idle"
    last_run_id: str = ""
    health_grade: str = "UNKNOWN"
    health_score: int = 0
    state_json: dict = None
    run_count: int = 0
    error_count: int = 0
    consecutive_errors: int = 0
    paused_at: Optional[str] = None
    paused_reason: Optional[str] = None
    updated_at: str = ""

    def __post_init__(self):
        if self.state_json is None:
            self.state_json = {}


def _headers() -> dict:
    return {
        "apikey": SUPABASE_SERVICE_KEY,
        "Authorization": f"Bearer {SUPABASE_SERVICE_KEY}",
        "Content-Type": "application/json",
        "Prefer": "return=representation",
    }


def get_loop_state(loop_name: str) -> Optional[LoopState]:
    """Get current state for a loop."""
    if not SUPABASE_URL:
        return None
    try:
        resp = requests.get(
            f"{SUPABASE_URL}/rest/v1/vcc_loop_state",
            headers=_headers(),
            params={"loop_name": f"eq.{loop_name}", "limit": "1"},
            timeout=10,
        )
        resp.raise_for_status()
        rows = resp.json()
        if not rows:
            return None
        row = rows[0]
        return LoopState(
            loop_name=row["loop_name"],
            status=row["status"],
            last_run_id=row.get("last_run_id", ""),
            health_grade=row.get("health_grade", "UNKNOWN"),
            health_score=row.get("health_score", 0),
            state_json=row.get("state_json", {}),
            run_count=row.get("run_count", 0),
            error_count=row.get("error_count", 0),
            consecutive_errors=row.get("consecutive_errors", 0),
            paused_at=row.get("paused_at"),
            paused_reason=row.get("paused_reason"),
            updated_at=row.get("updated_at", ""),
        )
    except Exception as e:
        log.error(f"Failed to get loop state: {e}")
        return None


def update_loop_state(
    loop_name: str,
    status: Optional[str] = None,
    last_run_id: Optional[str] = None,
    health_grade: Optional[str] = None,
    health_score: Optional[int] = None,
    state_json: Optional[dict] = None,
    increment_run: bool = False,
    increment_error: bool = False,
    reset_errors: bool = False,
    paused_reason: Optional[str] = None,
) -> bool:
    """Update loop state. Only updates fields that are provided."""
    if not SUPABASE_URL:
        return False

    update = {}
    if status is not None:
        update["status"] = status
        if status == "paused":
            update["paused_at"] = datetime.now(timezone.utc).isoformat()
            if paused_reason:
                update["paused_reason"] = paused_reason
        elif status in ("idle", "running"):
            update["paused_at"] = None
            update["paused_reason"] = None
    if last_run_id is not None:
        update["last_run_id"] = last_run_id
        update["last_run_at"] = datetime.now(timezone.utc).isoformat()
    if health_grade is not None:
        update["health_grade"] = health_grade
    if health_score is not None:
        update["health_score"] = health_score
    if state_json is not None:
        update["state_json"] = state_json

    if not update and not increment_run and not increment_error and not reset_errors:
        return True  # Nothing to update

    try:
        # Handle increments with RPC or direct update
        if increment_run:
            update["run_count"] = "run_count + 1"  # Handled via RPC
        if increment_error:
            update["error_count"] = "error_count + 1"
            update["consecutive_errors"] = "consecutive_errors + 1"
        if reset_errors:
            update["consecutive_errors"] = 0

        resp = requests.patch(
            f"{SUPABASE_URL}/rest/v1/vcc_loop_state",
            headers=_headers(),
            params={"loop_name": f"eq.{loop_name}"},
            json=update,
            timeout=10,
        )
        resp.raise_for_status()
        return True
    except Exception as e:
        log.error(f"Failed to update loop state: {e}")
        return False


def append_history(
    loop_name: str,
    run_id: str,
    status: str,
    health_grade: Optional[str] = None,
    health_score: Optional[int] = None,
    state_json: Optional[dict] = None,
    duration_ms: Optional[int] = None,
    triggered_by: Optional[str] = None,
) -> bool:
    """Append a history entry (audit trail)."""
    if not SUPABASE_URL:
        return False

    try:
        resp = requests.post(
            f"{SUPABASE_URL}/rest/v1/vcc_loop_state_history",
            headers=_headers(),
            json={
                "loop_name": loop_name,
                "run_id": run_id,
                "status": status,
                "health_grade": health_grade,
                "health_score": health_score,
                "state_json": state_json or {},
                "duration_ms": duration_ms,
                "triggered_by": triggered_by,
            },
            timeout=10,
        )
        resp.raise_for_status()
        return True
    except Exception as e:
        log.error(f"Failed to append history: {e}")
        return False
```

## BRIDGE-007: Loop Orchestrator


### Implementation — `tools/bridge/loop_orchestrator.py`

```python
#!/usr/bin/env python3
"""Loop Orchestrator — Coordinates VCC (CRP) and Perplexity loops.

Persistent NATS listener that:
1. Monitors both loop completion events
2. Maintains merged health snapshot
3. Enforces GO/NO-GO rules
4. Auto-pauses VCC on CRITICAL infrastructure
5. Auto-resumes VCC when infrastructure recovers

Usage: python tools/bridge/loop_orchestrator.py
"""

import asyncio
import json
import os
import logging
import time
from datetime import datetime, timezone, timedelta
from typing import Optional

import nats

from tools.bridge.bridge_nats_publisher import (
    BridgeNatsClient,
    CycleCompletedPayload,
    DiagnosticCompletedPayload,
    BuildPausePayload,
    BuildResumePayload,
)
from tools.bridge.bridge_health_schema import (
    CitadelHealthSnapshot,
    merge_snapshots,
    compute_overall_grade,
)
from tools.bridge.bridge_state_store import (
    get_loop_state,
    update_loop_state,
    append_history,
)
from tools.bridge.bridge_feedback_injection import evaluate_crp_impact

log = logging.getLogger("bridge.orchestrator")
log.setLevel(logging.INFO)

NATS_URL = os.environ.get("NATS_URL", "nats://localhost:4222")
STREAM_NAME = "VCC_BRIDGE"
ORCHESTRATOR_CONSUMER = "vcc-bridge-orchestrator"

# GO/NO-GO thresholds
CRITICAL_PAUSE_THRESHOLD = 2           # Pause VCC after N consecutive CRITICAL diagnostics
RECOVERY_REQUIRED_GRADE = "DEGRADING"  # Must reach at least this grade to auto-resume
MAX_PAUSE_HOURS = 6                     # Auto-resume after N hours regardless (deadman switch)
HEALTH_CHECK_INTERVAL_SECONDS = 300     # Check health every 5 minutes


class LoopOrchestrator:
    """Coordinates VCC and Perplexity loops via NATS events."""

    def __init__(self):
        self._nc: Optional[nats.NATS] = None
        self._js = None
        self._running = True
        self._nats_client = BridgeNatsClient(NATS_URL)

        # In-memory state (backed by vcc_loop_state DB)
        self._last_crp_snapshot: Optional[CitadelHealthSnapshot] = None
        self._last_pplx_snapshot: Optional[CitadelHealthSnapshot] = None
        self._consecutive_critical: int = 0
        self._vcc_paused: bool = False
        self._pause_time: Optional[float] = None

    async def start(self):
        """Start the orchestrator — connect to NATS and begin monitoring."""
        connect_opts = {"servers": [NATS_URL]}
        self._nc = await nats.connect(**connect_opts)
        self._js = self._nc.jetstream()

        log.info("Loop orchestrator starting...")

        # Update state store
        update_loop_state("orchestrator", status="running")

        # Subscribe to both completion events
        self._crp_sub = await self._js.pull_subscribe(
            subject="citadel.vcc.cycle.completed",
            durable=f"{ORCHESTRATOR_CONSUMER}-crp",
            stream=STREAM_NAME,
        )
        self._diag_sub = await self._js.pull_subscribe(
            subject="citadel.diagnostic.completed",
            durable=f"{ORCHESTRATOR_CONSUMER}-diag",
            stream=STREAM_NAME,
        )

        log.info("Subscribed to both loop completion events")

        # Run event loop
        while self._running:
            await self._poll_events()
            await self._check_deadman_switch()
            await asyncio.sleep(1)

    async def _poll_events(self):
        """Poll for new events from both subscriptions."""
        # Poll CRP events
        try:
            msgs = await self._crp_sub.fetch(batch=1, timeout=2)
            for msg in msgs:
                await self._handle_crp_event(msg)
        except nats.errors.TimeoutError:
            pass

        # Poll diagnostic events
        try:
            msgs = await self._diag_sub.fetch(batch=1, timeout=2)
            for msg in msgs:
                await self._handle_diagnostic_event(msg)
        except nats.errors.TimeoutError:
            pass

    async def _handle_crp_event(self, msg):
        """Process CRP cycle completion event."""
        try:
            data = json.loads(msg.data.decode())
            payload = CycleCompletedPayload(**data)
            log.info(f"[CRP] Cycle completed: {payload.cycle_id} ({payload.phase})")

            # Build CRP health snapshot
            from tools.bridge.bridge_health_schema import CodeHealth, ServiceHealth
            self._last_crp_snapshot = CitadelHealthSnapshot(
                snapshot_id=f"CRP-{payload.cycle_id}",
                source="crp",
                crp_cycle_id=payload.cycle_id,
                crp_version=payload.crp_version,
                guardrail_pass=payload.guardrail_pass,
                drift_clean=payload.drift_clean,
                sla_compliant=payload.sla_compliant,
                srs_codes_touched=payload.srs_codes_touched,
                code=CodeHealth(
                    test_passed=payload.test_passed,
                    test_failed=payload.test_failed,
                    test_count=payload.test_passed + payload.test_failed,
                ),
            )

            # Update state store
            update_loop_state(
                "crp_v2",
                status="idle",
                last_run_id=payload.cycle_id,
                health_grade="HEALTHY" if payload.guardrail_pass else "DEGRADING",
                state_json=data,
                reset_errors=True,
            )

            # Merge and evaluate
            await self._merge_and_evaluate()
            await msg.ack()

        except Exception as e:
            log.error(f"[CRP] Event handling failed: {e}")
            await msg.nak(delay=10)

    async def _handle_diagnostic_event(self, msg):
        """Process Perplexity diagnostic completion event."""
        try:
            data = json.loads(msg.data.decode())
            payload = DiagnosticCompletedPayload(**data)
            log.info(f"[PPLX] Diagnostic completed: {payload.diag_id} (grade: {payload.health_grade})")

            # Build Perplexity health snapshot
            from tools.bridge.bridge_health_schema import InfrastructureHealth, RevenueHealth
            self._last_pplx_snapshot = CitadelHealthSnapshot(
                snapshot_id=f"PPLX-{payload.diag_id}",
                source="perplexity",
                diag_id=payload.diag_id,
                perplexity_score=payload.health_score,
                l3_verdict=payload.l3_verdict,
                blockers=payload.blockers,
                recommendations=payload.recommendations,
                triggered_by=payload.triggered_by,
                overall_grade=payload.health_grade,
                overall_score=payload.health_score,
                infrastructure=InfrastructureHealth(
                    nats_connections=payload.nats_connections,
                ) if payload.nats_connections else None,
                revenue=RevenueHealth(
                    mrr_cents=payload.mrr_cents or 0,
                    active_subscriptions=payload.active_subscriptions or 0,
                ) if payload.mrr_cents else None,
            )

            # Track consecutive CRITICAL
            if payload.health_grade == "CRITICAL":
                self._consecutive_critical += 1
            else:
                self._consecutive_critical = 0

            # Update state store
            update_loop_state(
                "perplexity_v2",
                status="idle",
                last_run_id=payload.diag_id,
                health_grade=payload.health_grade,
                health_score=payload.health_score,
                state_json=data,
                reset_errors=payload.health_grade != "CRITICAL",
            )

            # Merge and evaluate
            await self._merge_and_evaluate()
            await msg.ack()

        except Exception as e:
            log.error(f"[PPLX] Event handling failed: {e}")
            await msg.nak(delay=10)

    async def _merge_and_evaluate(self):
        """Merge snapshots and enforce GO/NO-GO rules."""
        merged = merge_snapshots(self._last_crp_snapshot, self._last_pplx_snapshot)

        # Update orchestrator state
        update_loop_state(
            "orchestrator",
            health_grade=merged.overall_grade,
            health_score=merged.overall_score,
            state_json=merged.model_dump(),
        )

        log.info(f"[MERGED] Grade: {merged.overall_grade} | Score: {merged.overall_score}/100")

        # ── GO/NO-GO Rules ───────────────────────────────────────

        # Rule 1: Auto-pause VCC on consecutive CRITICAL diagnostics
        if (
            self._consecutive_critical >= CRITICAL_PAUSE_THRESHOLD
            and not self._vcc_paused
        ):
            await self._pause_vcc(
                f"Infrastructure CRITICAL for {self._consecutive_critical} consecutive diagnostics"
            )

        # Rule 2: Auto-resume VCC when infrastructure recovers
        if (
            self._vcc_paused
            and merged.overall_grade in ("HEALTHY", "DEGRADING")
            and self._consecutive_critical == 0
        ):
            await self._resume_vcc("Infrastructure recovered")

    async def _check_deadman_switch(self):
        """Auto-resume VCC if paused for too long (deadman switch)."""
        if (
            self._vcc_paused
            and self._pause_time
            and (time.time() - self._pause_time) > MAX_PAUSE_HOURS * 3600
        ):
            await self._resume_vcc(f"Deadman switch: paused for >{MAX_PAUSE_HOURS}h")

    async def _pause_vcc(self, reason: str):
        """Pause VCC build loop."""
        log.warning(f"[PAUSE] VCC paused: {reason}")
        self._vcc_paused = True
        self._pause_time = time.time()

        # Publish pause event
        await self._nats_client.connect()
        await self._nats_client.publish_build_pause(BuildPausePayload(
            reason=reason,
            triggered_by="orchestrator",
            health_grade=self._last_pplx_snapshot.overall_grade if self._last_pplx_snapshot else None,
        ))
        await self._nats_client.disconnect()

        # Update state store
        update_loop_state("crp_v2", status="paused", paused_reason=reason)

    async def _resume_vcc(self, reason: str):
        """Resume VCC build loop."""
        log.info(f"[RESUME] VCC resumed: {reason}")
        self._vcc_paused = False
        self._pause_time = None

        # Publish resume event
        await self._nats_client.connect()
        await self._nats_client.publish_build_resume(BuildResumePayload(
            reason=reason,
            triggered_by="orchestrator",
            health_grade=self._last_pplx_snapshot.overall_grade if self._last_pplx_snapshot else None,
        ))
        await self._nats_client.disconnect()

        # Update state store
        update_loop_state("crp_v2", status="idle")

    def stop(self):
        self._running = False


if __name__ == "__main__":
    logging.basicConfig(level=logging.INFO, format="%(asctime)s %(name)s %(levelname)s %(message)s")
    orchestrator = LoopOrchestrator()
    try:
        asyncio.run(orchestrator.start())
    except KeyboardInterrupt:
        orchestrator.stop()
        log.info("Orchestrator stopped by operator")
```

## BRIDGE-008: CRP v2.1 Schema Extensions


### Dataclass Updates — `tools/crp_v21_extensions.py`

```python
#!/usr/bin/env python3
"""CRP v2.1 Schema Extensions.

Extends the CycleReport dataclass from v2.0 with Phase 2.1 bridge fields.
Adds §7.4 (Perplexity Diagnostic), §7.5 (Bridge Status), §7.6 (Merged Health).

Apply these changes to vcc_cycle_reporter.py CycleReport dataclass.
"""

from dataclasses import dataclass, field
from typing import Optional


@dataclass
class PerplexityDiagnosticSection:
    """CRP §7.4 — Perplexity Diagnostic (last run)."""
    available: bool = False
    diag_id: str = ""
    health_grade: str = "UNKNOWN"
    health_score: int = 0
    top_blockers: list = field(default_factory=list)  # max 3
    infrastructure_status: str = "unknown"
    l3_verdict: str = ""
    source: str = "none"             # "state_store" | "notion_api" | "none"
    stale: bool = False
    timestamp: str = ""


@dataclass
class BridgeStatusSection:
    """CRP §7.5 — Bridge Status."""
    bridge_active: bool = False
    data_source: str = "none"        # "state_store" | "notion_api" | "none"
    last_diagnostic_age: str = ""    # ISO timestamp
    stale: bool = False
    nats_stream_healthy: bool = False
    state_store_healthy: bool = False
    pipeline_pass_rate: Optional[float] = None
    nats_connections: Optional[int] = None


@dataclass
class MergedHealthSection:
    """CRP §7.6 — Merged Health."""
    perplexity_grade: str = "UNKNOWN"
    perplexity_score: Optional[int] = None
    merged_grade: str = "UNKNOWN"
    merged_score: int = 0
    mrr_cents: Optional[int] = None
    active_subscriptions: Optional[int] = None
    recommendations: list = field(default_factory=list)  # max 3


@dataclass
class CycleReportV21Extensions:
    """Fields to add to CycleReport for CRP v2.1.

    Usage in vcc_cycle_reporter.py:
        @dataclass
        class CycleReport:
            # ... existing v2.0 fields ...
            crp_version: str = "2.1"  # bumped from 2.0

            # §7.4 — Perplexity Diagnostic
            perplexity_diagnostic: PerplexityDiagnosticSection = field(
                default_factory=PerplexityDiagnosticSection
            )
            # §7.5 — Bridge Status
            bridge_status: BridgeStatusSection = field(
                default_factory=BridgeStatusSection
            )
            # §7.6 — Merged Health
            merged_health: MergedHealthSection = field(
                default_factory=MergedHealthSection
            )

            # NATS event flag
            nats_event_published: bool = False
    """
    crp_version: str = "2.1"
    perplexity_diagnostic: PerplexityDiagnosticSection = field(
        default_factory=PerplexityDiagnosticSection
    )
    bridge_status: BridgeStatusSection = field(
        default_factory=BridgeStatusSection
    )
    merged_health: MergedHealthSection = field(
        default_factory=MergedHealthSection
    )
    nats_event_published: bool = False


# ── CRP v2.1 Validator Extension ─────────────────────────────────

CRP_V21_REQUIRED_SECTIONS = [
    "perplexity_diagnostic",
    "bridge_status",
    "merged_health",
]


def validate_v21_extensions(report: dict, strict: bool = False, cycle_number: int = 0) -> list[str]:
    """Validate CRP v2.1 extensions.

    Args:
        report: The CRP report dict
        strict: If True, missing sections are errors. If False, warnings.
        cycle_number: Current cycle count — grace period for first 3 cycles.

    Returns:
        List of error/warning strings.
    """
    issues = []
    version = report.get("crp_version", "")

    if version < "2.1":
        issues.append(f"VERSION: CRP version is {version}, expected ≥2.1 for bridge support")
        return issues

    # Check required sections
    for section in CRP_V21_REQUIRED_SECTIONS:
        if section not in report:
            if cycle_number <= 3 and not strict:
                issues.append(f"GRACE: §{section} missing (grace period: cycle {cycle_number}/3)")
            else:
                issues.append(f"MISSING: §{section} required in CRP v2.1")

    # Check NATS event published
    if not report.get("nats_event_published", False):
        issues.append("NATS: cycle completion event not published")

    # Check perplexity diagnostic if present
    pplx = report.get("perplexity_diagnostic", {})
    if pplx and pplx.get("available") and pplx.get("health_grade") == "CRITICAL":
        issues.append("CRITICAL: Perplexity reports CRITICAL — deployment blocked")

    return issues
```
